\documentclass[rgb]{standalone}
\usepackage{tkz-euclide}
\definecolor{myorange}{hsb}{0.0833, 1, 0.8}
\definecolor{mygreen}{hsb}{0.3333, 1, 0.8}
\definecolor{myblue}{hsb}{0.5833, 1, 0.8}
\definecolor{mymagenta}{hsb}{0.8333, 1, 0.8}
\begin{document}
	\begin{tikzpicture}[scale=1.5, font=\Large]
		% Coordinate system
		\tkzInit[xmin=-5,xmax=5,ymin=-2,ymax=2]
		\tkzGrid[color=lightgray]
		\tkzDrawX[thick]
		\tkzDrawY[thick]
		\draw[very thick,domain=-450:450, smooth, samples=500, variable=\x,myblue] plot ({\x/90}, {cos(\x)});
		\draw[very thick,domain=-450:450, smooth, samples=500, variable=\x,myorange] plot ({\x/90}, {sin(\x)});
		\node[below=0.5mm] at (2,0){$\pi$};		
		\node[below=0.5mm] at (4,0){$2\pi$};
		\node[below=0.5mm] at (-2,0){$-\pi$};	
		\node[below=0.5mm] at (-4,0){$-2\pi$};
		\node[left=0.5mm] at (0,1){$1$};
		\node[left=0.5mm] at (0,-1){$-1$};
		\node[left,myblue] at (-0.35,1){$y=\cos(x)$};
		\node[right,myorange] at (1.4,1){$y=\sin(x)$};
	\end{tikzpicture}	
\end{document}
