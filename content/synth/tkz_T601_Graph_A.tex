\definecolor{myorange}{hsb}{0.0833, 1, 0.8}

\definecolor{myblue}{hsb}{0.5833, 1, 0.8}
\definecolor{mymagenta}{hsb}{0.8333, 1, 0.8}
\begin{document}
\begin{tikzpicture}[scale=1.5, font=\Large]
	% Coordinate system
	\tkzInit[xmin=-3,xmax=3,ymin=-1,ymax=5]
	\tkzGrid[color=lightgray]
	\tkzDrawX[thick,label=$x$]
	\tkzDrawY[thick,label=$y$]
	% Graph
	\draw[very thick,domain={-sqrt(5)}:{sqrt(5)}, smooth, variable=\x, myblue] plot ({\x}, {\x*\x});
	\draw[dashed] (2,4) -- (0,4);
	\draw[dashed] (2,0) -- (2,4);
	\draw[dashed] (-1,1) -- (0,1);
	\draw[dashed] (-1,0) -- (-1,1);
	\node[very thick, draw, circle,fill,myblue,inner sep=2] at (-1,1) {};
	\node[very thick, draw, circle,fill,myblue,inner sep=2] at (0,0) {};
	\node[very thick, draw, circle,fill,myblue,inner sep=2] at (2,4) {};
	\node[left, myblue] at (-1,1){$(-1;1)$};
	\node[right, myblue] at (2,4){$(2;4)$};
	\node[below, myblue] at (0,0){$(0;0)$};
	% Labels
	\node[below=0.5mm] at (-2,0){$-2$};
	\node[below=0.5mm] at (2,0){$2$};
