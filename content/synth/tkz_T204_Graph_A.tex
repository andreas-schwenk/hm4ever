\documentclass[rgb]{standalone}
\usepackage{tkz-euclide}
\definecolor{myorange}{hsb}{0.0833, 1, 0.8}
\definecolor{mygreen}{hsb}{0.3333, 1, 0.8}
\definecolor{myblue}{hsb}{0.5833, 1, 0.8}
\definecolor{mymagenta}{hsb}{0.8333, 1, 0.8}
\begin{document}
\begin{tikzpicture}[scale=1.5, font=\Large]
	% Coordinate system
	\tkzInit[xmin=-3,xmax=3,ymin=-2,ymax=4]
	\tkzGrid[color=lightgray]
	\tkzDrawX[thick,label=$x$]
	\tkzDrawY[thick,label=$y$]
	\draw[very thick,domain={-sqrt(5)}:{sqrt(5)}, smooth, variable=\x, myblue] plot ({\x}, {\x*\x-1});
	\draw[very thick,domain=-2:2, smooth, variable=\x, myorange] plot ({\x}, {\x*\x});
	\draw[very thick,domain={-sqrt(2)}:{sqrt(2)}, smooth, variable=\x, mygreen] plot ({\x}, {\x*\x+2});
	% Labels
	\node[below=0.5mm] at (-2,0){$-2$};
	\node[below=0.5mm] at (2,0){$2$};
	\node[left=0.5mm] at (0,-1){$-1$};
	\node[left=0.5mm] at (0,1){$1$};
	\node[left=0.5mm] at (0,3){$3$};
	\node[anchor=center, myorange] at (0,1.3){$f_1(x)=x^2$};
	\node[anchor=center, myblue] at (0,-1.25){$f_3(x)=x^2-1$};
	\node[anchor=center, mygreen] at (0,3.75){$f_3(x)=x^2+2$};
\end{tikzpicture}	
\end{document}
