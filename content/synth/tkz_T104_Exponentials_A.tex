\documentclass[rgb]{standalone}
\usepackage{tkz-euclide}
\definecolor{myorange}{hsb}{0.0833, 1, 0.8}
\definecolor{mygreen}{hsb}{0.3333, 1, 0.8}
\definecolor{myblue}{hsb}{0.5833, 1, 0.8}
\definecolor{mymagenta}{hsb}{0.8333, 1, 0.8}
\begin{document}
	\begin{tikzpicture}[scale=1.5, font=\Large]
		% Coordinate system
		\tkzInit[xmin=-3,xmax=3,ymin=-0,ymax=6]
		\tkzGrid[color=lightgray]

		\tkzDrawY[thick,label=$y$]
		% Graphs
		\draw[very thick,domain={-3}:{ln(6)/ln(10)}, smooth, variable=\x, myblue] plot ({\x}, {10^(\x)});
		\draw[very thick,domain={-ln(6)/ln(10)}:{3}, smooth, variable=\x, myblue, densely dashed] plot ({\x}, {0.1^(\x)});	
		\draw[very thick,domain={-3}:{ln(6)/ln(3)}, smooth, variable=\x, myorange] plot ({\x}, {3^(\x)});
		\draw[very thick,domain={-ln(6)/ln(3)}:{3}, smooth, variable=\x, myorange, densely dashed] plot ({\x}, {(1/3)^(\x)});
		\draw[very thick,domain={-3}:{ln(6)/ln(2)}, smooth, variable=\x, mygreen] plot ({\x}, {2^(\x)});
		\draw[very thick,domain={-ln(6)/ln(2)}:{3}, smooth, variable=\x, mygreen, densely dashed] plot ({\x}, {(1/2)^(\x)});
		% Labels
		\node[below=0.5mm] at (0,0){$0$};
		\node[below=0.5mm] at (-2,0){$-2$};

		\node[above,myblue, rotate=90] at (-0.75,5.25){$y=\bigl(\frac{1}{10}\bigr)^x$};
		\node[below,myorange, rotate=90] at (1.55,5.15){$y=3^x$};
		\node[above,myorange, rotate=90] at (-1.55,5.15){$y=\bigl(\frac{1}{3}\bigr)^x$};
		\node[below,mygreen, rotate=90] at (2.425,5.15){$y=2^x$};
		\node[above,mygreen, rotate=90] at (-2.4,5.15){$y=\bigl(\frac{1}{2}\bigr)^x$};
