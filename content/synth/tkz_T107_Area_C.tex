\documentclass[rgb]{standalone}
\usepackage{tkz-euclide}
\definecolor{myorange}{hsb}{0.0833, 1, 0.8}
\definecolor{mygreen}{hsb}{0.3333, 1, 0.8}
\definecolor{myblue}{hsb}{0.5833, 1, 0.8}
\definecolor{mymagenta}{hsb}{0.8333, 1, 0.8}
\begin{document}
	\begin{tikzpicture}[scale=1.5, font=\Large]
		% Coordinate system
		\tkzInit[xmin=-4,xmax=2,ymin=-3,ymax=3]
		\tkzGrid[color=lightgray]
		\tkzDrawX[thick]
		\tkzDrawY[thick]	
		\draw[fill,draw=none,myblue!50!myorange, fill opacity=0.4,domain=-3:1,variable=\x]  plot ({\x}, {2*cos(45*\x)}) -- plot ({\x}, {2*sin(45*\x)});
		\draw[very thick,domain=-4:2, smooth,samples=200,variable=\x,myblue] plot ({\x}, {2*cos(45*\x)});	
		\draw[very thick,domain=-4:2, smooth,samples=200,variable=\x,myorange] plot ({\x}, {2*sin(45*\x)});			
		% Labely
		\node[myblue,left] at (-0.5,2){$y=\cos(x)$};
		\node[myorange,below] at (-2,-2){$y=\sin(x)$};
		\node[below=0.5mm] at (-3,0){$-\frac{3\pi}{4}$};
		\node[below=0.5mm] at (-1,0){$-\frac{\pi}{4}$};
		\node[below=0.5mm] at (1,0){$\frac{\pi}{4}$};
		\node[left=0.5mm] at (0,2){$1$};
		\node[left=0.5mm] at (0,-2){$-1$};
	\end{tikzpicture}	
\end{document}
