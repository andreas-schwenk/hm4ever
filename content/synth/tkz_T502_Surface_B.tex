\documentclass[rgb]{standalone}
\usepackage{pgfplots}
\usepgfplotslibrary{patchplots}
\pgfplotsset{colormap/viridis}
\definecolor{myorange}{hsb}{0.0833, 1, 0.8}
\definecolor{myblue}{hsb}{0.5833, 1, 0.8}
\begin{document}
	\begin{tikzpicture}[font=\Large]
		\begin{axis}[scale=1.5,thick,grid=both, minor tick num =1,view={45}{45},
			xmin=-3, xmax=3, ymin=-3,ymax=3, zmin=0, zmax=2, 
			ztick={0,1,2}, 
			xlabel=$x_1$, ylabel=$x_2$, zlabel=$f(x_1 {,} x_2)$]
			\addplot3[surf, patch, domain=-3:3, y domain=-3:3,
			samples=20] {(x^2+2*y^2)*exp(1-x^2-y^2)};
			\draw[line width=2.5pt, -latex, myblue] (axis cs:1,-1,{3*exp(-1)}) -- (axis cs:0,-1,{7*exp(-1)});
			\draw[line width=2.5pt, -latex, myorange] (axis cs:1,-1,{3*exp(-1)}) -- (axis cs:1,0,{5*exp(-1)});
			\node[myblue] at (axis cs:0,-1,{0.15+7*exp(-1)}) {$D_{(-1,0)} f(1,-1)$};
			\node[myorange] at (axis cs:1,0.5,{0.05+5*exp(-1)}) {$D_{(0,1)} f(1,-1)$};
		\end{axis}
	\end{tikzpicture}
\end{document}
