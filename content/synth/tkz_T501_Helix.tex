\begin{document}

		\begin{axis}[scale=1.5,thick,grid=both,minor tick num =1,view={45}{45}, xmin=-1.5,xmax=1.5, ymin=-1.5,ymax=1.5, zmin=0, zmax=8,ztick={0,4,8}, zticklabels={$0$,$4\pi$,$8\pi$},
