\lang{de}{Untersuchen Sie, für welche Werte von $a,b\in\R$ die Funktion $f\colon\R\rightarrow\R$ mit
\[f(x)=\begin{cases}
        ae^{bx},&\text{falls } x<0,\\
        1+x ,&\text{falls }x \geq 0,\\
       \end{cases}\]
im Nullpunkt stetig ist.}

\lang{en}{Decide for which values $a,b\in\R$ the function $f\colon\R\rightarrow\R$ with
\[f(x)=\begin{cases}
        ae^{bx},&\text{if } x<0,\\
        1+x ,&\text{if }x \geq 0,\\
       \end{cases}\]
is continuous in $x=0$.}

\begin{tabs*}[\initialtab{0}\class{exercise}]

  \tab{
  \lang{de}{Antwort }
  \lang{en}{Answer}}
	\[ a= 1, b\in \R.\]
  \tab{
  \lang{de}{Lösung }
  \lang{en}{Solution}}
  
  \lang{de}{Da $f_{1}(x)=ae^{bx}$ und $f_{2}(x)=1+x$ auf $\R$ stetig sind, 
muss der rechtsseitige und linksseitige Limes für $x$ gegen $0$ 
übereinstimmen. Wir dürfen $x=0$ in beide Komponentenfunktionen einsetzen.
Es gilt nun $f_{1}(0)=a$ und $f(0)=f_{2}(0)=1$. Also ist $f$ genau für $a=1$ und alle $b\in\R$ stetig.}
\lang{en}{Since $f_{1}(x)=ae^{bx}$ and $f_{2}(x)=1+x$ are continuous in all $x \in \R$,
the function is continuous if and only if the right-handed and left-handed limits
are equal in $x = 0$. We have $f_{1}(0)=a$ and $f(0)=f_{2}(0)=1$.
Therefore, $f$ is continuous if and only if $a = 1$, with any $b \in R$.
}
