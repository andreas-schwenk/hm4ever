	\begin{question}
    \lang{de}{\explanation{Lösen Sie die Gleichung $y=\var{f}$ nach $x$ auf.}}
      \lang{en}{\explanation{Solve the equation $y=\var{f}$ for $x$. }}
		\begin{variables}
			\randint[Z]{a}{-10}{10}
			\randint{b}{-10}{10}
			\function[calculate]{c}{a/2}
			\function[calculate]{d}{b/2}
			\function[expand,normalize]{f}{c*x+d}
			\function{l}{(y-d)/c}
		\end{variables}
		\type{input.function}
		\field{rational}
		\lang{de}{\text{Bestimmen Sie auf einem geeigneten Bereich die Umkehrfunktion der Funktion $f: \mathbb{R} \to \mathbb{R}, \ f(x) = \var{f}$ und 
           geben Sie die Funktionsvorschrift in der Variablen $y$ ein.}}
    \lang{en}{\text{Determine, on a suitable domain, the inverse function of the function $f: \mathbb{R} \to \mathbb{R}, \ f(x) = \var{f}$ and 
           enter it as a function of the variable $y$.}}
   

		\begin{answer}
			\text{$f^{-1}(y)=$}
			\solution{l}
			\checkAsFunction{y}{-10}{10}{100}
		\end{answer}
	\end{question}
