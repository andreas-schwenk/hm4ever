	\randomquestionpool{1}{2}
      \begin{question}
      
      \begin{variables}
      		\randint{a}{-3}{3}
      		\randint[Z]{b}{-2}{2}
      		\randint{c}{1}{4}
      		\randint[Z]{d}{-2}{2}
      		\function[expand,normalize]{f}{x^2+a*x+b}
      		\function[expand,normalize]{g}{c*x+d}
      		\function[expand,normalize]{fg}{(c*x+d)^2+a*(c*x+d)+b}
      		\function[expand,normalize]{gf}{c*(x^2+a*x+b)+d}
      \end{variables}

      \type{input.function}
      \field{real}
     \lang{de}{\text{Bestimmen Sie die Kompositionen $f\circ g$ und $g\circ f$ der Funktionen\\
      $f:\R\to \R, \  f(x) = \var{f}$ und $g:\R\to \R, \  g(x) = \var{g}$.\\
      Multiplizieren Sie alle Terme aus.}}
      \lang{en}{\text{Determine the compositions $f\circ g$ and $g\circ f$ of the functions\\
      $f:\R\to \R, \ f(x) = \var{f}$ and $g:\R\to \R, \ g(x) = \var{g}$.\\
      Multiply all terms out.}}

      \begin{answer}
          \text{$(f\circ g)(x)=$}
          \solution{fg}
		  \allowForInput[false]{)}
          \checkAsFunction{x}{-10}{10}{10}
          \lang{de}{\explanation{$(f\circ g)(x)=f(g(x))$. Man muss also in die Vorschrift für $f$ statt $x$ den Ausdruck für $g(x)$ einsetzen. 
          Multiplizieren Sie anschließend aus.}}
          \lang{en}{\explanation{$(f\circ g)(x)=f(g(x))$. So we have to substitute the expression for $g(x)$ into the definition of $f$ in place of $x$. 
          Then multiply out.}}
      \end{answer}
      
      \begin{answer}
		 \text{$(g\circ f)(x)=$}
          \solution{gf}
		  \allowForInput[false]{)}
          \checkAsFunction{x}{-10}{10}{10}
          \lang{de}{\explanation{$(g\circ f)(x)=g(f(x))$. Man muss also in die Vorschrift für $g$ statt $x$ den Ausdruck für $f(x)$ einsetzen. 
          Multiplizieren Sie anschließend aus.}}
           \lang{en}{\explanation{$(g\circ f)(x)=g(f(x))$. So we have to substitute the expression for $f(x)$ into the definition of $g$ in place of $x$. 
          Then multiply out.}}
      \end{answer}
  \end{question} 


  \begin{question}
  \lang{de}{\explanation{$(T\circ S)(n)=T(S(n))$. Man muss also in die Vorschrift für $T$ statt $x$ den Ausdruck für $S(n)$ einsetzen.}} 
  \lang{en}{\explanation{$(T\circ S)(n)=T(S(n))$. Therefore, we substitute the expression for $S(n)$ into the definition of $T$ in place of $x$.}}
  \begin{variables}
  	\randint{x1}{0}{2}
  	\randint{x2}{3}{5}
  	\randint{x3}{6}{7}
  	\randint{a}{2}{4}
  	\randint{b}{1}{3}
  	\randint{m}{1}{2}
	\function[calculate]{d}{a*x3+b+m}  
	\function[calculate]{d1}{d+1}  
	\function[normalize]{sn}{a*n+b}  
	\function{ts}{d1-sn}  
	\substitute{y1}{ts}{n}{x1}  
	\substitute{y2}{ts}{n}{x2}  
	\substitute{y3}{ts}{n}{x3}  
  \end{variables}

      \type{input.number}
      \field{real}
      \lang{de}{\text{Gegeben seien die Funktionen\\
      $S:\{\var{x1};\var{x2};\var{x3} \}\to \{1;\ldots ; \var{d}\}, \ S(n) = \var{sn}$ und\\
      $T:\{1;\ldots ; \var{d}\}\to \{1;\ldots ; \var{d}\}, \ T(x) = \var{d1}-x$.\\
      Bestimmen Sie die Funktionswerte der zusammengesetzten Abbildung $T\circ S$.}}     
      \lang{en}{\text{Given the functions\\
      $S:\{\var{x1},\var{x2},\var{x3} \}\to \{1,\ldots , \var{d}\}, \ S(n) = \var{sn}$ and\\
      $T:\{1,\ldots , \var{d}\}\to \{1,\ldots , \var{d}\}, \ T(x) = \var{d1}-x$,\\
      determine the values of the composition $T\circ S$.}}  
      
      \begin{answer}
      \text{$(T\circ S)(\var{x1})=$}
      \solution{y1}
      \end{answer}
      
      \begin{answer}
      \text{$(T\circ S)(\var{x2})=$}
      \solution{y2}
      \end{answer}
      
      \begin{answer}
      \text{$(T\circ S)(\var{x3})=$}
      \solution{y3}
      \end{answer}
  \end{question}
