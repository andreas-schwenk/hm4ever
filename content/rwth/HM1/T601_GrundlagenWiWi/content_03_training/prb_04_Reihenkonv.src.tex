%Aufgabe 1

\begin{question}
    \type{mc.multiple}
    \field{rational}
    \lang{de}{\text{Entscheiden Sie, welche der folgenden Aussagen wahr sind.}}
    \lang{en}{\text{Decide which of the following statements are true.}}
    \begin{choice}
       	\lang{de}{\text{(a) $\sum_{k=\var{k}}^{\infty}{(-\var{a})^k}$ ist absolut konvergent.}}
        \lang{en}{\text{(a) $\sum_{k=\var{k}}^{\infty}{(-\var{a})^k}$ is absolutely convergent.}}
		\solution{false}
        \lang{de}{\explanation[NOT[equalChoice(00000)] AND equalChoice(1???0)]{(a) ist falsch. 
            Die geometrische Reihe $\sum_{k=0}^\infty q^k$ ist divergent für 
            $| q| \geq 1$.}}
        \lang{en}{\explanation[NOT[equalChoice(00000)] AND equalChoice(1???0)]{(a) is false. 
            The geometric series $\sum_{k=0}^\infty q^k$ is divergent for 
            $| q| \geq 1$.}}
     \end{choice}	
     %%
	\begin{choice}
        \lang{de}{\text{(b) $\sum_{n=0}^{\infty}\frac{(-\var{c})^n}{n!}$ konvergiert absolut.}}
        \lang{en}{\text{(b) $\sum_{n=0}^{\infty}\frac{(-\var{c})^n}{n!}$ converges absolutely.}}
		\solution{true}
        \lang{de}{\explanation[NOT[equalChoice(00000)] AND equalChoice(?0??0)]{(b) stimmt. Das ist die Reihe $\exp(-\var{c})$. 
            Die  Exponentialreihe konvergiert überall absolut.}}
        \lang{en}{\explanation[NOT[equalChoice(00000)] AND equalChoice(?0??0)]{(b) is correct. This is the series $\exp(-\var{c})$. 
            The exponential series converges absolutely everywhere.}}
     \end{choice}
     %%
	\begin{choice}
        \lang{de}{\text{(c) Es gilt $\sum_{n\in\N_0}\left(\frac{1}{\var{a}}\right)^{n}=\var{f}$. }}
        \lang{en}{\text{(c) $\sum_{n\in\N_0}\left(\frac{1}{\var{a}}\right)^{n}=\var{f}$. }}
		\solution{true}
        \lang{de}{\explanation[NOT[equalChoice(00000)] AND equalChoice(??0?0)]{(c) stimmt. Geometrische Reihe! }}
        \lang{en}{\explanation[NOT[equalChoice(00000)] AND equalChoice(??0?0)]{(c) is true. Geometric series! }}
     \end{choice}
     %%
	\begin{choice}
        \lang{de}{\text{(d) Es gilt $\sum_{k=1}^{\infty}{\left(-\frac{1}{2}\right)^k}=\var{g}$.}}
        \lang{en}{\text{(d) $\sum_{k=1}^{\infty}{\left(-\frac{1}{2}\right)^k}=\var{g}$.}}
		\solution{false}
        \lang{de}{\explanation[NOT[equalChoice(00000)] AND equalChoice(???10)]{(d) ist falsch. 
            Als geometrische Reihe ohne $0$-tes Glied ist der Wert 
            $\frac{1}{1+\frac{1}{2}}-1=-\frac{1}{3}$.}}
        \lang{en}{\explanation[NOT[equalChoice(00000)] AND equalChoice(???10)]{(d) is wrong. 
            As a geometric series with no $0$-th member, the value is 
            $\frac{1}{1+\frac{1}{2}}-1=-\frac{1}{3}$.}}
     \end{choice}
     %%
     \begin{choice}
     \lang{de}{\text{(e) Keine der Aussagen ist wahr.}}
     \lang{en}{\text{(e) None of the statements are true.}}
     \solution{false}
     \lang{de}{\explanation[equalChoice(????1)]{Nicht alle angegebenen Aussagen sind falsch. Erinnern Sie sich an bekannte Reihen.}}
     \lang{en}{\explanation[equalChoice(????1)]{Not all of the statements are false. Think of familiar series.}}
     \end{choice}
\end{question}
