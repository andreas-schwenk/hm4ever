\begin{problem}
    \begin{variables}
        \randint{a}{1}{5}
        \function[calculate]{quadrat}{a^2}
        \function[calculate]{mixed}{2*a}
        \function[expand]{f}{e^(2*x)-quadrat}
        \function[expand]{f1}{e^(2*x)-mixed*e^x+quadrat}
        \function{sol}{ln(a)}
    \end{variables}
    \begin{question}
        \type{input.number}
        \lang{de}{\text{Berechnen Sie die Nullstellen der Funktion $f(x)=\var{f1}$:}}
        \lang{en}{\text{Calculate the zeros of the function $f(x)=\var{f1}$:}}
        \begin{answer}
          \lang{de}{\text{Runden Sie Ihr Ergebnis bitte auf 2 Nachkommastellen: $x=$}}
          \lang{en}{\text{Please round your result to 2 decimal places: $x=$}}
            \solution{sol}
        \end{answer}
        \lang{de}{\explanation{Es handelt sich hierbei um die zweite binomische Formel:\\
        $f(x)=(e^x-\var{a})^2$. Also ist $x=\ln(\var{a})$}}
         \lang{en}{\explanation{This is the second binomial formula:\\
        $f(x)=(e^x-\var{a})^2$. So $x=\ln(\var{a})$}}
    \end{question}
\end{problem}
