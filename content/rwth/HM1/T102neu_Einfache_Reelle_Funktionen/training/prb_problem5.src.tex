\begin{problem}
%
  \begin{question} 
      \text{Wie lautet die Geradengleichung der Geraden, die durch den Punkt 
        $P = (\var{p1x} ; \var{p1y})$ verl"auft und die Steigung $m= \var{m}$ besitzt?}
%       
	\begin{variables}
       	\randint{p1x}{-10}{10}
      	\randint{p1y}{-10}{10}
       	\randint[Z]{m}{-10}{10}       	   	
       	\function[calculate]{b}{p1y-(m*p1x)}  
       	\function[normalize]{f}{m * x+b}
       	\function[normalize]{g}{y - p1y}
       	\function{h}{m * (x - p1x)}           	        	
    \end{variables}
%    
	\type{input.function}
  	\field{rational}
%    
	\begin{answer}
        \text{$g: \; y=$}
        \solution{f}
        \checkAsFunction{x}{-10}{10}{100} 
        
        \explanation{$\,m= \var{m}$ und $\, P = (\var{p1x} ; \var{p1y}), \,$
                    eingesetzt in die Geradengleichung $\, y= mx+b, \,$ liefern den 
                    fehlenden Parameter $\,b.$}
     \end{answer}
%    
\end{question}
	
\end{problem}
