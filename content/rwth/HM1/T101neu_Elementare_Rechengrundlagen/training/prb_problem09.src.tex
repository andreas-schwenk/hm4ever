% Q1 - Tautologie

    \begin{question}
      \text{Welche der folgenden Formeln mit Variablen $A$, $B$ und $C$ 
      sind immer wahr, unabhängig von den jeweiligen Wahrheitswerten der
      einzelnen Variablen? Man nennt eine solche Formel auch eine \textit{Tautologie.}}
      
      \explanation{}

      \permutechoices{1}{4}
      \type{mc.multiple}
      \field{real}
      \precision{3}
      
      \begin{pool}
        \begin{variables}
          \string{a}{A}
          \string{b}{B}
          \string{c}{C}
        \end{variables}
        \begin{variables}
          \string{a}{A}
          \string{b}{C}
          \string{c}{B}
        \end{variables}
        \begin{variables}
          \string{a}{B}
          \string{b}{A}
          \string{c}{C}
        \end{variables}
        \begin{variables}
          \string{a}{B}
          \string{b}{C}
          \string{c}{A}
        \end{variables}
        \begin{variables}
          \string{a}{C}
          \string{b}{A}
          \string{c}{B}
        \end{variables}
        \begin{variables}
          \string{a}{C}
          \string{b}{B}
          \string{c}{A}
        \end{variables}
      \end{pool}

      \begin{choice}
        \text{$\var{a}\vee (\var{c}\wedge \neg \var{c})$}
        \solution{false}  
      \end{choice}
      \begin{choice}
        \text{$\neg (A\wedge C) \vee \neg B \vee (A \wedge B)$}
        \solution{true}  
      \end{choice}
      \begin{choice}
        \text{$\neg (\var{b} \wedge \neg (\var{b}\vee \var{a}))$}
        \solution{true}  
      \end{choice}
      \begin{choice}
        \text{$(\var{a}\wedge \var{c})\vee \neg (\var{a}\wedge \var{c})$}
        \solution{true}  
      \end{choice}
    \end{question}

    \begin{question}
      \text{Welche der folgenden Formeln mit Variablen $A$, $B$ und $C$ 
      sind niemals wahr bzw. immer falsch, unabhängig von den jeweiligen Wahrheitswerten der

      \permutechoices{1}{4}

      \field{real}

      \begin{pool}
        \begin{variables}

        \end{variables}

      \begin{choice}

      \begin{choice}
        \text{$(\var{a}\wedge \var{c})\vee \neg (\var{a}\wedge \var{c})$}

      \begin{choice}

        \solution{true}
      \end{choice}
