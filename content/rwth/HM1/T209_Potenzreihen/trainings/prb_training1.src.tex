\begin{question}

	
\begin{variables}
	\randint[Z]{q}{-9}{9}
	\function[calculate]{sol}{1/abs(q)}
	
\end{variables}

	\type{input.number}
	\field{rational} 
	\precision{3}
    \lang{de}{
	    \text{
	     Bestimmen Sie den Konvergenzradius der Potenzreihe
        $\sum_{k=0}^{\infty} ( \var{q} \cdot z )^{k}.$\\
        %Geben Sie die Antwort auf drei Nachkommastellen genau an.
    }}
    
    \begin{answer}
    \text{Der Konvergenzradius ist gegeben durch r =}
	    \solution{sol}
      \explanation{Der Konvergenzradius der geometrischen Reihe $\sum_{k=0}^\infty z^k$ ist $1$.
    Daher ist für $b\neq 0$ der Konvergenzradius der  Reihe $\sum_{k=0}^\infty (bz)^k$ gleich $r=\frac{1}{\vert b\vert}$.}    
    \end{answer}
    
\end{question}

\begin{question}
	
\begin{variables}
	\randint[Z]{q}{2}{5}
    \randint[Z]{a}{-3}{3}
	\function[calculate]{sol}{(1/q)^a}
	\function[normalize]{f}{a*k}
\end{variables}

	\type{input.number}
	\field{rational} 
	\precision{3}
    \lang{de}{
	    \text{
	     Bestimmen Sie den Konvergenzradius der Potenzreihe
        $\sum_{k=0}^{\infty} \var{q}^{\var{f}} \cdot z^{k}.$\\
        %Geben Sie die Antwort auf drei Nachkommastellen genau an.
    }}
    
    \begin{answer}
    \text{Der Konvergenzradius ist gegeben durch r =}
	    \solution{sol}
    \explanation{Der Konvergenzradius der geometrischen Reihe $\sum_{k=0}^\infty z^k$ ist $1$.
    Daher ist für $a\neq 0$ der Konvergenzradius der  Reihe $\sum_{k=0}^\infty (az)^k$ gleich $r=\frac{1}{\vert a\vert}$. 
    Man muss die Reihe allerdings vorher in diese Form bringen.}
    \end{answer}
    
\end{question}
