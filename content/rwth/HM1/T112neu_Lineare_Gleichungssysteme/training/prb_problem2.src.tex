	\begin{question} 
        \lang{de}{\text{Geben Sie die Lösung des folgenden linearen Gleichungssystems an.\\
        Ist das Gleichungssystem nicht eindeutig lösbar, setzen Sie bitte die von Ihnen gewählte frei\\ wählbare Variable gleich $t$, damit Ihre Lösung vom System richtig korrigiert werden kann.\\ Verwenden Sie bitte außerdem Brüche und keine gerundeten Werte.\\
        \begin{align*}
        \var{a}x&+\var{b}y&\var{c}z& \ = \ &\var{j}\\
        &\var{d1}y&+\var{f}z& \ = \ &\var{k}\\
        &&\var{i1}z& \ = \ &\var{l}
        \end{align*}
        }}
        \lang{en}{\text{Find the solution of the following linear system.\\
        If the linear system is not uniquely solvable, set the parameter you introduce for one of the variables equal to $t$ so that the system can check your answer properly. When inputting your answer use fractions instead of rounded values.\\
        \begin{align*}
        \var{a}x&+\var{b}y&\var{c}z& \ = \ &\var{j}\\
        &\var{d1}y&+\var{f}z& \ = \ &\var{k}\\
        &&\var{i1}z& \ = \ &\var{l}
        \end{align*}
        }}
        \lang{de}{\explanation{Bestimmen Sie die Lösung durch Rückwärtseinsetzen.}}
        \lang{en}{\explanation{Find the solution via back substitution.}}
        \type{input.function}
        \field{rational}
        
        \begin{variables}
            \randint[Z]{a}{-10}{10}
            \randint[Z]{b}{0}{10}
            \randint[Z]{c}{-10}{0}
            \randint[Z]{d1}{-10}{10}
            \randint[Z]{f}{0}{10}
            \randint[Z]{i1}{-10}{10}
            \randadjustIf{i1}{i1=1}                                                                  
            
            \randint{r}{-5}{5}
            \randint{s}{-5}{5}
            \randint{t}{-5}{5}
                        
            \function[calculate]{j}{a*r+b*s+c*t}
            \function[calculate]{k}{d1*s+f*t} 
            \function[calculate]{l}{i1*t}                        
                   
            \function[normalize]{lsg1}{r}
            \function[normalize]{lsg2}{s}
            \function[normalize]{lsg3}{t}         
        \end{variables}
	
		\begin{answer}
              \text{$x=$}
              \solution{lsg1}
              \inputAsFunction{t}{u}
        \end{answer}
        
        \begin{answer}
        	  \text{$y=$}
              \solution{lsg2} 
              \inputAsFunction{t}{v}
              
        \end{answer}
        
        \begin{answer}
        	  \text{$z=$}
              \solution{lsg3} 
              \inputAsFunction{t}{w}
              
              \checkFuncForZero{abs(a*u[x]+b*v[x]+c*w[x]-j)+abs(d1*v[x]+f*w[x]-k)+abs(i1*w[x]-l)}{-10}{10}{100} 
        \end{answer}
        
	\end{question}
	
	\begin{question} 
        \lang{de}{\text{Geben Sie die Lösung des folgenden linearen Gleichungssystems an.\\
        Ist das Gleichungssystem nicht eindeutig lösbar, setzen Sie bitte die von Ihnen gewählte frei\\ wählbare Variable gleich $t$, damit Ihre Lösung vom System richtig korrigiert werden kann.\\ Verwenden Sie bitte außerdem Brüche und keine gerundeten Werte.\\

        &&0& \ = \ &0
        \end{align*}
        }}
        \lang{en}{\text{Find the solution of the following linear system.\\
        If the linear system is not uniquely solvable, set the parameter you introduce for one of the variables equal to $t$ so that the system can check your answer properly. When inputting your answer use fractions instead of rounded values.\\
        \begin{align*}
        \var{a}x&+\var{b}y&\var{c}z& \ = \ &\var{j}\\
        &\var{d1}y&+\var{f}z& \ = \ &\var{k}\\
        &&0& \ = \ &0
        \end{align*}
        }}
        \lang{de}{\explanation{Setzen Sie eine der Variablen $y$ oder $z$ gleich einem Parameter $t$ und bestimmen Sie anschließend durch Rückwärtseinsetzen die weiteren Variablen.}}
        \lang{en}{\explanation{Set one of the variables $y$ or $z$ equal to a parameter $t$ and find the other variables via back substitution.}}
        \type{input.function}
        \field{rational}
        
        \begin{variables}
            \randint[Z]{a}{-5}{5}
            \randint[Z]{b}{0}{10}
            \randint[Z]{c}{-5}{0}
            \randint[Z]{d1}{-5}{0}
            \randint[Z]{f}{0}{5}
   
            \randint{r}{-5}{5}
            \randint{s}{-5}{5}
            \randint{q}{-5}{5}
            
            \function[calculate]{j}{a*r+b*s+c*q}
            \function[calculate]{k}{d*r+d1*s+f*q} 
                      
            \function[normalize]{lsg1}{(j*d1-b*k)/(d1*a)+t*(b*f-d1*c)/(d1*a)}
            \function[normalize]{lsg2}{k/d1-t*f/d1}
            \function{lsg3}{t}     
        \end{variables}
        
        \begin{answer}
              \text{$x=$}
              \solution{lsg1}
              \inputAsFunction{t}{u}
        \end{answer}
        
        \begin{answer}
        	  \text{$y=$}
              \solution{lsg2} 
              \inputAsFunction{t}{v}
              
        \end{answer}
        
        \begin{answer}
        	  \text{$z=$}
              \solution{lsg3} 
              \inputAsFunction{t}{w}
              
              \checkFuncForZero{abs(a*u[x]+b*v[x]+c*w[x]-j)+abs(d1*v[x]+f*w[x]-k)+abs((x-u[x])*(x-v[x])*(x-w[x]))}{-10}{10}{100} 
        \end{answer}
        
	\end{question}
