    \begin{variables}

        \end{case}
        \begin{default}
          \function{maxv1}{v12a}
        \end{default}
    \end{switch}
    \function[calculate]{v1-1n}{abs(v11)+abs(v12)}
    \function[calculate]{v1-2n}{sqrt((v11)*(v11)+(v12)*(v12))}
    
    
    \randint[Z]{v21}{-6}{6}

    \end{switch}
    \function[calculate]{v}{max1v2}
    \begin{switch}
        \begin{case}{v>=v23a}
          \function{maxv2}{v}
        \end{case}
        \begin{default}
          \function{maxv2}{v23a}
        \end{default}
    \end{switch}
    \function[calculate]{v2-1n}{abs(v21)+abs(v22)+abs(v23)}
    \function[calculate]{v2-2n}{sqrt((v21)*(v21)+(v22)*(v22)+(v23)*(v23))}
    
    
    \randint[Z]{a11}{-8}{12}
    \randint[Z]{a12}{-12}{0}
    \randint[Z]{a21}{-15}{6}
    \randint[Z]{a22}{-12}{8}
    \function[calculate]{a1j}{abs(a11)+abs(a12)}
    \function[calculate]{a2j}{abs(a21)+abs(a22)}
    \begin{switch}
        \begin{case}{a1j>=a2j}

    \lang{de}{\text{Berechnen Sie fÃ¼r die folgenden Vektoren und Matrizen die angegebenen Normen.\\
    $v_1 :=\begin{pmatrix}\var{v11}\\\var{v12}\end{pmatrix}$, $v_2 :=\begin{pmatrix}\var{v21}\\\var{v22}\\ \var{v23} \end{pmatrix}$,
    $A:=\begin{pmatrix}
    \var{a11} & \var{a12} \\
    \var{a21} & \var{a22} 
    \end{pmatrix}$ und $B:=\begin{pmatrix}
    \var{b11} & \var{b12} & \var{b13} \\
    \var{b21} & \var{b22} & \var{b23}\\
    \var{b31} & \var{b32} & \var{b33}
    \end{pmatrix}$ .}}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert v_1\Vert_{\infty}$=}
    \solution{maxv1}
    \explanation{$\Vert v_1\Vert_\infty=\max_{1\leq i\leq 2}\vert v_{1i}\vert
    =max\{\vert v_{11}\vert;\vert v_{12}\vert\}$.}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert v_1\Vert_{1}$=}
    \solution{v1-1n}
    \explanation{$\Vert v_1\Vert_1=\sum_{i=1}^2\vert v_{1i}\vert=\vert v_{11}\vert+\vert v_{12}\vert$.}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert v_1\Vert_{2}$=}
    \solution{v1-2n}
    \explanation{${\Vert v_1 \Vert_2}=\sqrt{v_{11}^2+v_{12}^2}$.}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert v_2\Vert_{\infty}$=}
    \solution{maxv2}
    \explanation{$\Vert v_2\Vert_\infty=\max_{1\leq i\leq 3}\vert v_{1i}\vert$.}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert v_2\Vert_{1}$=}
    \solution{v2-1n}
    \explanation{$\Vert v_2\Vert_1=\sum_{i=1}^3\vert v_{2i}\vert$.}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert v_2\Vert_{2}$=}
    \solution{v2-2n}
    \explanation{${\Vert v_2 \Vert_2}=\sqrt{v_{21}^2+v_{22}^2+v_{23}^2}$.}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert A\Vert_{\infty}$=}
    \solution{ainfn}
    \explanation{$\Vert A\Vert_{\infty}=\max_{i=1,2}{\sum_{j=1}^{2}\vert a_{ij}\vert}
    =\max\{\vert a_{11}\vert+\vert a_{12}\vert;\vert a_{21}\vert+\vert a_{22}\vert\}.$}
    \end{answer}
    
    \begin{answer}
    \type{input.number}
    \text{$\Vert A\Vert_{1}$=}
    \solution{a1n}
    \explanation{$\Vert A\Vert_{1}=\max_{j=1,2}{\sum_{i=1}^{2}\vert a_{ij}\vert}
    =\max\{\vert a_{11}\vert+\vert a_{21}\vert;\vert a_{12}\vert+\vert a_{22}\vert\}.$}
    \end{answer}

    \end{answer}

    \end{answer}
