\begin{problem}
    \begin{variables}
    \randint[Z]{a}{-8}{12}
    \randint[Z]{b}{-12}{12}
    \randint[Z]{c}{-15}{10}
    \randint[Z]{d}{2}{8}
    \randint[Z]{e}{9}{19}
    \randint[Z]{f}{-12}{12}
    \randint[Z]{g}{1}{15}
    \randint[Z]{h}{2}{8}
    \randint[Z]{k}{-8}{8}
    \randint[Z]{i}{-8}{12}
    \randint[Z]{j}{-8}{12}
    \randint[Z]{l}{-15}{-1}
	\end{variables}
        
    \randomquestionpool{1}{3}

	\begin{question}        
        \type{mc.multiple}	
        
          \text{Seien $(a_{n})_{n\in\N},(b_{n})_{n\in\N}\subset\R$ und $(x^{(n)})_{n\in\N},(y^{(n)})_{n\in\N}\subset\R^{k}$ konvergente 
          Folgen. Welche der unten stehenden Folgen sind konvergent?}
          
          \explanation{$(-1)^{n}$ und $\sin({\frac{n\pi}{2}})$ sind nicht konvergent. Alle anderen Folgen 
            sind konvergent.}
		\begin{choice}
			\text{$\begin{pmatrix}\frac{\var{a}n^2+2}{n^2+2n+3}\\ b_{n}\end{pmatrix}$,}
			\solution{true}
        \end{choice}
		\begin{choice}
			\text{$\begin{pmatrix}(-1)^{n}\\ a_{n}\end{pmatrix}$,}
			\solution{false}
		\end{choice}
        \begin{choice}
			\text{$-y^{(n)}+x^{(n)}$,}
			\solution{true}
		\end{choice}
        \begin{choice}
			\text{$\begin{pmatrix}\sin({\frac{n\pi}{2}})\\\frac{n+1}{n^3+1}\end{pmatrix}$,}
			\solution{false}
		\end{choice}
        \begin{choice}
			\text{$\begin{pmatrix}\var{f}\cdot (\frac{\var{h}}{\var{e}})^n\\
                                      \var{k}\\\var{k}\end{pmatrix}$.}
			\solution{true}
		\end{choice}
     \end{question}
     
     
	\begin{question}        
        \type{mc.multiple}	
        
          \text{Seien $(a_{n})_{n\in\N},(b_{n})_{n\in\N}\subset\R$ und $(x^{(n)})_{n\in\N},(y^{(n)})_{n\in\N}\subset\R^{k}$ konvergente 
          Folgen. Welche der unten stehenden Folgen sind konvergent?}
          
          \explanation{$n$ und $\begin{pmatrix}n+1\\ n+2\\\vdots\\ n+k\end{pmatrix}$ sind nicht konvergent. Alle anderen Folgen 
            sind konvergent.}
		\begin{choice}
			\text{$\begin{pmatrix}n+1\\ n+2\\\vdots\\ n+k\end{pmatrix}+y^{(n)}$, wenn 
            $\begin{pmatrix}n+1\\ n+2\\\vdots\\ n+k\end{pmatrix}\in\R^k$,}
			\solution{false}
        \end{choice}
		\begin{choice}
			\text{$\begin{pmatrix}0\\ a_{n}\\ b_{n}\end{pmatrix}$,}
			\solution{true}
		\end{choice}
        \begin{choice}
			\text{$\begin{pmatrix}\frac{\var{b}}{\var{d}^n}\\ 5b_{n}\end{pmatrix}$,}
			\solution{true}
		\end{choice}
        \begin{choice}
			\text{$\begin{pmatrix} a_{n}\\ n\\ b_{n}\end{pmatrix}$,}
			\solution{false}
		\end{choice}
        \begin{choice}
			\text{$3x^{(n)}-y^{(n)}$.}
			\solution{true}
		\end{choice}
     \end{question}
     
     
	\begin{question}        
        \type{mc.multiple}	
        
          \text{Seien $(a_{n})_{n\in\N},(b_{n})_{n\in\N}\subset\R$ und $(x^{(n)})_{n\in\N},(y^{(n)})_{n\in\N}\subset\R^{k}$ konvergente 
          Folgen. Welche der unten stehenden Folgen sind konvergent?}
          
            \explanation{$\cos({n\pi+\frac{\pi}{3}})$ ist nicht konvergent. Alle anderen Folgen 
            sind konvergent.}
		\begin{choice}
			\text{$\begin{pmatrix}\frac{\var{g}}{(\var{l})^n}\\ 5b_{n}\end{pmatrix}$,}
			\solution{true}
        \end{choice}
		\begin{choice}
			\text{$\begin{pmatrix}\cos({n\pi+\frac{\pi}{3}})\\\frac{n+1}{n^3+1}\end{pmatrix}$,}
			\solution{false}
		\end{choice}
        \begin{choice}
			\text{$\begin{pmatrix}a_{n}\\ 3\\ 3a_{n}+b_{n}+2\end{pmatrix}$,}
			\solution{true}
		\end{choice}
        \begin{choice}
			\text{$\begin{pmatrix}a_{n}\\ \vdots\\ a_{n}\end{pmatrix}+y^{(n)}$, wenn $\begin{pmatrix}a_{n}\\ \vdots\\ a_{n}\end{pmatrix}
            \in\R^k$,}
			\solution{true}
		\end{choice}
        \begin{choice}
			\text{$-3x^{(n)}-\frac{3n+1}{n+5}y^{(n)}$.}
			\solution{true}
		\end{choice}
     \end{question}	
	
	
\end{problem}
