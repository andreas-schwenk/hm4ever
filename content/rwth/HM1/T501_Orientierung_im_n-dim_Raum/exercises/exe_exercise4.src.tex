\title{
  \lang{en}{Exercise 4}
    \lang{de}{Ü04: Normen}}
    
Berechnen Sie die folgenden Normen und Abstände.

\begin{enumerate}
\item[(a)] $\Vert\left(\begin{smallmatrix}-4\\1\end{smallmatrix}\right)\Vert_{\infty}$, 
$\Vert\left(\begin{smallmatrix}2\\7\\1\end{smallmatrix}\right)\Vert_{\infty}$ und
der Abstand zwischen $\left(\begin{smallmatrix}6\\3\\1\end{smallmatrix}\right)$ und
$\left(\begin{smallmatrix}4\\6\\7\end{smallmatrix}\right)$ mit der Maximumsnorm.

\item[(b)] $\Vert\left(\begin{smallmatrix}3\\5\end{smallmatrix}\right)\Vert_{2}$, 
$\Vert\left(\begin{smallmatrix}2\\-4\\ -\frac{1}{2}\end{smallmatrix}\right)\Vert_{2}$ und
der Abstand zwischen $\left(\begin{smallmatrix}5\\\frac{1}{3}\\2\end{smallmatrix}\right)$ und
$\left(\begin{smallmatrix}4\\-1\\\frac{1}{5}\end{smallmatrix}\right)$ mit der Euklidischen Norm.

\item[(c)] $\Vert\left(\begin{smallmatrix}\frac{-6}{5}\\2\end{smallmatrix}\right)\Vert_{1}$, 
$\Vert\left(\begin{smallmatrix}-3\\-2\\1\end{smallmatrix}\right)\Vert_{1}$ und
der Abstand zwischen $\left(\begin{smallmatrix}6\\6\\1\end{smallmatrix}\right)$ und
$\left(\begin{smallmatrix}4\\-4\\4\end{smallmatrix}\right)$ mit der 1-Norm.
\end{enumerate}

  \begin{tabs*}[\initialtab{0}\class{exercise}]
    
    \tab{
      Lösung (a)
    }
    \[\Vert \left(\begin{smallmatrix}-4\\1\end{smallmatrix}\right)\Vert_{\infty}
    ={\max\{\vert -4\vert;\vert 1\vert\}}=4.\]
    \[\Vert \left(\begin{smallmatrix}2\\7\\1\end{smallmatrix}\right)\Vert_{\infty}
    ={\max\{\vert 2\vert;\vert 7\vert;\vert 1\vert\}}=7.\]
    Der Abstand zwischen den Vektoren $\left(\begin{smallmatrix}6\\3\\1\end{smallmatrix}\right)$ und
    $\left(\begin{smallmatrix}4\\6\\7\end{smallmatrix}\right)$ ist
    \[\Vert \left(\begin{smallmatrix}6\\3\\1\end{smallmatrix}\right)-\left(\begin{smallmatrix}4\\6\\7\end{smallmatrix}\right)\Vert_{\infty}
    =\Vert \left(\begin{smallmatrix}2\\-3\\-6\end{smallmatrix}\right)\Vert_{\infty}
    ={\max\{\vert 2\vert;\vert -3\vert;\vert -6\vert\}}=6.\]
    
    \tab{
    Lösung (b)
    }
    \[\Vert \left(\begin{smallmatrix}3\\5\end{smallmatrix}\right)\Vert_{2}
    =\sqrt{ 3^{2}+ 5^{2}}=\sqrt{ 34}.\]
    \[\Vert \left(\begin{smallmatrix}2\\-4\\ -\frac{1}{2}\end{smallmatrix}\right)\Vert_{2}
    =\sqrt{ 2^{2}+ (-4)^{2}+(-\frac{1}{2})^{2}}=\frac{9}{2}.\]
    
    Der Abstand zwischen den Vektoren $\left(\begin{smallmatrix}5\\\frac{1}{3}\\2\end{smallmatrix}\right)$ und
    $\left(\begin{smallmatrix}4\\-1\\\frac{1}{2}\end{smallmatrix}\right)$ ist
    \[\Vert \left(\begin{smallmatrix}5\\\frac{1}{3}\\2\end{smallmatrix}\right)-\left(\begin{smallmatrix}4\\-1\\\frac{1}{2}\end{smallmatrix}\right)\Vert_{2}
    =\Vert \left(\begin{smallmatrix}1\\\frac{4}{3}\\ \frac{3}{2}\end{smallmatrix}\right)\Vert_{2}
    =\sqrt{ 1^{2}+ (\frac{4}{3})^{2}+(\frac{3}{2})^{2}}=\frac{\sqrt{181}}{6}.\]

    \tab{
    Lösung (c)
    }
    \[\Vert \left(\begin{smallmatrix}\frac{-6}{5}\\2\end{smallmatrix}\right)\Vert_{1}
    ={\vert \frac{-6}{5}\vert+\vert 2\vert}=\frac{16}{5}.\]
    \[\Vert \left(\begin{smallmatrix}-3\\-2\\1\end{smallmatrix}\right)\Vert_{1}
    ={\vert -3\vert+\vert -2\vert+\vert 1\vert}=6.\]
    
    Der Abstand zwischen den Vektoren $\left(\begin{smallmatrix}6\\6\\1\end{smallmatrix}\right)$ und
    $\left(\begin{smallmatrix}4\\-4\\4\end{smallmatrix}\right)$ ist
    
    \[\Vert \left(\begin{smallmatrix}6\\6\\1\end{smallmatrix}\right)-\left(\begin{smallmatrix}4\\-4\\4\end{smallmatrix}\right)\Vert_{1}
    =\Vert \left(\begin{smallmatrix}2\\10\\-3\end{smallmatrix}\right)\Vert_{1}
    ={\vert 2\vert+\vert 10\vert+\vert -3\vert}=15.\]

 \end{tabs*}
