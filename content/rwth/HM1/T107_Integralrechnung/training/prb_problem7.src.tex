	\begin{question}
    \type{input.generic}
	
       \begin{variables}
       		\randint{ahilf}{2}{5}
       		\randint{bhilf}{1}{2}
       		\function[calculate]{a}{(-1)^bhilf*ahilf}
        	\function[calculate]{z}{a^2+1}
			\function[calculate]{y}{a^2-1}    
       \end{variables}

       

      
      

	
%begin-cosh
	   \lang{de}{
	     \text{ Welcher der hier gegebenen Funktionsterme gehört zu einer Stammfunktion
	     der Funktion $f$ mit $f(x)=\sin(x) \cdot e^{\var{a}x} \, $? \\
%ende-cosh	     
	     \textit{Tipp: Ableitungen bestimmen.}

	     }
	   }
	    \lang{en}{
        \text{
          Which of the following terms defines an antiderivative of the function $f$ where $f(x)=\sin(x) \cdot e^{\var{a}x} \, $? \\
          \textit{Hint: Calculate the derivatives of each option.}
        }
      }
      \begin{answer}
	   %\permutechoices{1}{6}
	   \type{mc.multiple}

	   \begin{choice}
	    \text{$ \frac{1}{\var{z}} e^{\var{a}x} (\var{a}\sin(x)-\cos(x))$ }
	    \solution{true}
	   \end{choice}

	   \begin{choice}
	    \text{$ \frac{1}{\var{z}} e^{\var{a}x} (\var{a}\sin(x)+\cos(x))$ }
	    \solution{false}
	   \end{choice}

	   \begin{choice}
	    \text{$ \frac{1}{\var{y}} e^{\var{a}x} (\var{a}\sin(x)-\cos(x))$ }
	    \solution{false}
	   \end{choice}

	   \begin{choice}
	    \text{$ \frac{1}{\var{y}} e^{\var{a}x} (\var{a}\sin(x)+\cos(x))$ }
	    \solution{false}
	   \end{choice}

	   \begin{choice}
	    \text{$ \frac{1}{\var{a}} e^{\var{a}x} (\var{a}\sin(x)+\cos(x))$ }
	    \solution{false}
	   \end{choice}

	   \begin{choice}
	    \text{$ \frac{1}{\var{a}} e^{\var{a}x} \cdot \cos(x)$ }
	    \solution{false}
	   \end{choice}
