\begin{enumerate}

\item[a)]
  \lang{de}{Die folgende Funktion $f(x)$ ist gegeben:}
  \lang{en}{The following function $f(x)$ is given:}

\[ f(x)= \frac{\sin(\pi x)}{x}  \]

  \lang{de}{Bestimmen Sie eine Stammfunktion von $f''(x)$ auf dem Intervall $(0;\infty)$. }
  \lang{en}{Determine the antiderivative of $f''(x)$ over the interval $(0, \infty)$.}

\item[b)]
  \lang{de}{Berechnen Sie anschließend $\big \int_1^{2} f''(x)\, dx$.}
  \lang{en}{Then calculate $\big \int_1^{2} f''(x)\, dx$.}
   \\



\end{enumerate}

\begin{tabs*}[\initialtab{0}\class{exercise}]

 \tab{\lang{de}{Antworten}
 \lang{en}{Answers}
  }

 \begin{enumerate}
\item[a)] $f'(x)= \frac{\pi \cos(\pi x)}{x} - \frac{\sin(\pi x)}{x^2}$
    \lang{de}{ist eine Stammfunktion von $f''(x)$.}
    \lang{en}{is an antiderivative of $f''(x)$.}
\item[b)] $\big\int_1^2 f''(x)\, dx = \frac{3}{2} \pi .$

\end{enumerate}
\tab{\lang{de}{L"osung zu a)}
\lang{en}{Solution for a)}
}

 \begin{incremental}[\initialsteps{1}]
  \step
    \lang{de}{Gesucht ist eine Funktion, deren Ableitung $f''(x)$ ist.}
    \lang{en}{We are looking for a function whose derivative is $f''(x)$.}
  \step
    \lang{de}{Die Funktion $f'(x)$ hat die Ableitung $f''(x)$ und ist daher eine Stammfunktion von $f''(x)$.}
    \lang{en}{The function $f'(x)$ has derivative $f''(x)$ and thus is an antiderivative of $f''(x)$.}
  \step
    \lang{de}{Die Ableitung $f'(x)$ von $f(x) = \sin(\pi x) \cdot \frac{1}{x}$ wird mit Hilfe der Produkt- und Kettenregel bestimmt:}
    \lang{en}{We find the derivative $f'(x)$ of $f(x) = \sin(\pi x) \cdot \frac{1}{x}$
    using the product and chain rule:}
  \step \[ f'(x) = \pi \cos(\pi x) \cdot \frac{1}{x} + \sin(\pi x)  \cdot \left( \frac{-1}{x^2} \right) =
  \frac{\pi \cos(\pi x)}{x} - \frac{\sin(\pi x)}{x^2} \]
    \lang{de}{Alternativ kann man auch die Quotientenregel zur Ableitung verwenden.}
    \lang{en}{Alternatively, we also could have used the quotient rule.}
  \end{incremental}

 \tab{\lang{de}{L"osung zu b)}
 \lang{en}{Solution for b)}
 }

 \begin{incremental}[\initialsteps{1}]
  \step
    \lang{de}{Die Funktion $f''(x)$ ist stetig und daher integrierbar auf
  dem Intervall $[1;2]$. Nach dem Hauptsatz der Differential- und Integralrechnung gilt }
  \lang{en}{The function $f(x)$ is continuous and thus integrable over the interval $[1, 2]$.
  By the fundamental theorem of calculus we have}
  \[ \int_1^2 f''(x)\, dx = \Big[ f'(x) \Big]_1^2 . \]
  \step
    \lang{de}{Außerdem gilt}
    \lang{en}{Also}
  \[ f'(x)= \frac{\pi \cos(\pi x)}{x} - \frac{\sin(\pi x)}{x^2} . \]
  \step
    \lang{de}{Also ist das Integral gleich }
    \lang{en}{Thus the integral is equal to}
  \[ \left[  \frac{\pi \cos(\pi x)}{x} - \frac{\sin(\pi x)}{x^2} \right]_1^2 =
  \frac{\pi}{2} - (- \pi) = \frac{3}{2} \pi . \]
  \end{incremental}

 \end{tabs*}
