\lang{de}{Berechnen Sie die folgenden Integrale:}

\begin{enumerate}[alph]

\item $\int_{-1}^2 (2x+1) dx$,
\item $\int_0^1 \frac{1}{x-3} dx$,
\item $\int_0^{\frac{\pi}{2}} \cos(2x)dx$,
\item $\int_2^{-1} (2x+1) dx$,
\item $\int_{-1}^1 e^{-|x|}dx$,
\item $\int_{-1}^1 x \cdot \left|x\right| dx$.

\end{enumerate}
