\begin{problem}
\begin{question}
		
		\begin{variables}
\randint{l1}{-2}{4}
\randint{k0}{1}{4}
\randint{k1}{1}{4}
\function[calculate]{l2}{l1+k0}
\function[calculate]{l3}{l1-k1}
\randint{a1}{-2}{2}
\randint{a2}{-1}{1}
\randint{b1}{-2}{2}
\randint{b2}{-1}{1}
\function{a3}{1}
\randadjustIf{b1}{b1^2+b2^2=0}
\function[calculate]{b3}{-a1*b1-a2*b2}
\function[calculate]{c1}{a2*b3-a3*b2}
\function[calculate]{c2}{a3*b1-a1*b3}
\function[calculate]{c3}{a1*b2-a2*b1}
\function[calculate]{n1}{a1^2+a2^2+a3^2}
\function[calculate]{n2}{b1^2+b2^2+b3^2}
\function[calculate]{n3}{n1*n2}
\matrix[calculate]{aa}{(a1*a1*l1)/n1+(b1*b1*l2)/n2+(c1*c1*l3)/n3 & (a1*a2*l1)/n1+(b1*b2*l2)/n2+(c1*c2*l3)/n3 & (a1*a3*l1)/n1+(b1*b3*l2)/n2+(c1*c3*l3)/n3  \\
(a2*a1*l1)/n1+(b2*b1*l2)/n2+(c2*c1*l3)/n3 & (a2*a2*l1)/n1+(b2*b2*l2)/n2+(c2*c2*l3)/n3 &
(a2*a3*l1)/n1+(b2*b3*l2)/n2+(c2*c3*l3)/n3 \\
(a3*a1*l1)/n1+(b3*b1*l2)/n2+(c3*c1*l3)/n3 & (a3*a2*l1)/n1+(b3*b2*l2)/n2+(c3*c2*l3)/n3 &
(a3*a3*l1)/n1+(b3*b3*l2)/n2+(c3*c3*l3)/n3 }
\matrix[calculate]{vv}{a1\\ a2\\ a3}
\matrix[calculate]{ww}{b1\\ b2\\ b3}
		\end{variables}
		
		\type{input.generic}
        \field{rational}
		\lang{de}{
	    	\text{Die Matrix $A=\var{aa}$ hat die Eigenvektoren $v_1=\var{vv}$ und $v_2=\var{ww}$.\\
Bestimmen Sie die Eigenwerte von $A$ und einen Eigenvektor von $A$, welcher zu $v_1$ und $v_2$
 linear unabhängig ist.\\
 Die Menge der Eigenwerte ist \ansref \\
Ein zu $v_1$ und $v_2$ linear unabhängiger Eigenvektor von $A$ ist: $($ \ansref  ,  \ansref ,  \ansref $)^T$}
		}
\lang{en}{
	    	\text{The matrix $A=\var{aa}$ has the eigenvectors $v_1=\var{vv}$ and $v_2=\var{ww}$.\\
Determine the eigenvalues of $A$ and an eigenvector of $A$, which is linearly independent of $v_1$ and $v_2$.\\
 The set of eigenvalues is \ansref \\
An eigenvector of $A$ linearly independent of $v_1$ and $v_2$ is: $($ \ansref  ,  \ansref ,  \ansref $)^T$}
		}


	    \begin{answer}
            \type{input.finite-number-set}
			\solution{l1,l2,l3}
            \lang{de}{\explanation{Die Menge der Eigenwerte ist nicht korrekt bestimmt. 
            Wenn Sie Eigenvektoren kennen, benötigen Sie dafür nicht das charakteristische Polynom.}}
	   \lang{en}{\explanation{The set of eigenvalues is not determined correctly.
    If you know the eigenvectors, you do not need the characteristic polynomial.}}
    \end{answer}    
	    \begin{answer}
            \type{input.function}
			\solution{c1}
			\inputAsFunction{x}{w1}
	    \end{answer}    
	    \begin{answer}
            \type{input.function}
			\solution{c2}
			\inputAsFunction{x}{w2}
	    \end{answer}   
	    \begin{answer}
            \type{input.function}
			\solution{c3}
			\inputAsFunction{x}{w3}
			\checkFuncForZero{(w1*c2-w2*c1)^2+(w1*c3-w3*c1)^2+1-|sign(w1^2+w2^2+w3^2)|}{-1}{1}{10}
	       \lang{en}{\explanation{The third eigenvector is not determined correctly. It must be perpendicular to the other vectors.}} 
         \lang{de}{\explanation{Der dritte Eigenvektor ist nicht korrekt bestimmt. Er muss senkrecht auf den ersten beiden stehen.}}
        \end{answer}   
	    
	\end{question}
