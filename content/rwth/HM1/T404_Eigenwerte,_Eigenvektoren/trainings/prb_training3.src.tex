Pool aus 3 Aufgaben. Die  $(4\times 4)$-Matrizen haben dabei besondere Form:
erste Spalte der Form $(*,0,0,0)^T$, letzte Spalte der Form $(0,0,0,*)$, oder Blockmatrix.

			\function[calculate]{l}{l0+1}

			\matrix[calculate]{aa}{v & y & z & w \\ 0 & (a*c-b)*(d*f*(1-l) + g*l)+ a*d*(1-l) + 1  & c*d*f*l - c*d*f - c*g*l + d*l - d  & -d*f*l + d*f + g*l\\
					  0 & a^2*(-c*d*f*l + c*d*f + c*g*l - d*l + d) + a*b*d*f*l - a*b*d*f - a*b*g*l + a*c*f - a*l + a - b*f  & a*c*d*f*l - a*c*d*f - a*c*g*l + a*d*l - a*d - c*f + l &  -a*d*f*l + a*d*f + a*g*l + f\\ 
					  0 & -a*b*c*d*f*l + a*b*c*d*f + a*b*c*g*l - a*b*d*l + a*b*d + a*c^2*f + a*c + b^2*d*f*l - b^2*d*f - b^2*g*l - b*c*f - b*l & b*c*d*f*l - b*c*d*f - b*c*g*l + b*d*l - b*d - c^2*f - c  & -b*d*f*l + b*d*f + b*g*l + c*f + l + 1}
			\function[expand,normalize]{s}{(v-t)*(1-t)*(l-t)*(1+l-t)}
		\end{variables}
