\title{
\lang{de}{Ü06a: Konvergenz}
\lang{en}{Exercise 6a}
}
 Untersuchen Sie, ob die folgenden Reihen konvergieren.
\begin{table}[\class{items}]
 \nowrap{a) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{1}{k^3}}$} &
 \nowrap{b) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{k+2}{k^2+4k-1}}$} &
 \nowrap{c) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{k^2+2}{k^4+3k}}$} \\
 \nowrap{d) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{k-3}{k+5}}$} &
 \nowrap{e) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{k}{2^k}}$} &
 \nowrap{f) $\displaystyle \ \sum_{k=1}^{\infty}{k^2\cdot 0,8^k}$}\\
 \nowrap{g) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{1,2^k}{k^4}}$}&
 \nowrap{h) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{k^3}{0,5^k}}$} &
 \nowrap{i) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{1}{\sqrt{k}}}$} \\
 \nowrap{j) $\displaystyle \ \sum_{k=1}^{\infty}{\left(1-\frac{1}{k^2}\right)}$} &
 \nowrap{k) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{2^k-k}{3^k+1}}$} &
 \nowrap{l) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{2^k+1}{k\cdot 2^k}}$} 
 
\end{table}
