\lang{de}{Untersuchen Sie, ob die folgenden Reihen konvergieren.}
\begin{table}[\class{items}]
\nowrap{a) $\displaystyle \ \sum_{k=1}^{\infty}{\left(1+\frac{1}{k}\right)^{k}}$} \\
\nowrap{b) $\displaystyle \ \sum_{k=1}^{\infty}{\frac{1}{5k}}$}
\end{table}


\begin{tabs*}[\initialtab{0}\class{exercise}]

\tab{\lang{de}{    Antworten    }}
    \lang{de}{ \begin{enumerate}[a)]
\item a) Die Reihe divergiert.
\item b) Die Reihe divergiert.
\end{enumerate} }


  \tab{
  \lang{de}{Lösung a)}
  }
\lang{de}{Die Folge $(a_{k})_{k\in\N}$ mit $a_{k}:=\left(1+\frac{1}{k}\right)^{k}$ ist keine Nullfolge, ihr Grenzwert ist $e$. Also ist die Reihe nicht konvergent.}

\tab{
\lang{de}{Lösung b)}}
\lang{de}{Angenommen, die Reihe wäre konvergent. Dann wäre nach den \ref[content_24_reihen_und_konvergenz][Grenzwertregeln]{rul:grenzwert-regeln} auch die Reihe 
\[5\cdot \sum_{k=1}^{\infty}{\frac{1}{5k}}=\sum_{k=1}^{\infty}{\frac{1}{k}}\]
konvergent. Dies ist ein Widerspruch zu  der Tatsache, dass die \ref[content_24_reihen_und_konvergenz][harmonische Reihe]{ex:erste-reihen} divergiert. Also war die Annahme der Konvergenz der Reihe falsch und die Reihe ist divergent.}

\end{tabs*}
