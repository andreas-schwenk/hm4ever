\lang{de}{Gegeben ist der Graph einer Funktion $f$. Bestimmen Sie das Vorzeichen der Ableitung von $f$ in geeigneten Intervallen.}
\lang{en}{Consider the graph of a function $f$. Determine the sign of the derivative of $f$ in suitable intervals.}

  \begin{incremental}[\initialsteps{1}]
  \step \lang{de}{Auf $(-\infty;x_0)$ ist $f$ streng monoton wachsend, deswegen ist das Vorzeichen von $f'$ dort positiv.}
  \lang{en}{$f$ is monotonically increasing in the interval $(-\infty,x_0)$, so the sign of $f'$ is positive there.}
  \step \lang{de}{In $x_0$ ist die Tangente an den Graphen horizontal, deshalb ist $f'(x_0)=0$.}
  \lang{en}{The tangent on the graph is horizontal at $x_0$, so $f'(x_0)=0$. }
  \step \lang{de}{Auf $(x_0 ;0)$ ist $f$ streng monoton fallend. Damit ist das Vorzeichen von $f'$ dort negativ.}
  \lang{en}{$f$ is monotonically decreasing in the interval $(x_0 ,0)$, so the sign of $f'$ is negative there.}
  \step \lang{de}{In $x=0$ ist wieder $f'(0)=0$, da die Tangente an den Graphen die Steigung $0$ hat.}
  \lang{en}{$f'(0)=0$ again at $x=0$, since the tangent on the graph has a slope of $0$.}
  \step \lang{de}{Auf $(0;\infty)$ steigt $f$ monoton. Damit hat $f'$ dort ein positives Vorzeichen. }
  \lang{en}{$f$ increases monotonically in the interval $(0,\infty)$. Therefore $f'$ has a positive sign there.  }
   
 
\lang{de}{Insgesamt erh\"{a}lt man die folgende Tabelle: 

\begin{table}
 Bereich   &$(-\infty;x_0)$&$x_0$&$(x_0 ;0)$&$0$&$(0;\infty)$\\
$f'$&$\;\;\;\;\; +$&$\;0$&$\;\;\;-$&$0$&$\;\;\;+$\\
\end{table} }  

\lang{en}{Altogether we obtain the following table:

\begin{table}
 Interval   &$(-\infty,x_0)$&$x_0$&$(x_0 ,0)$&$0$&$(0,\infty)$\\
$f'$&$\;\;\;\;\; +$&$\;0$&$\;\;\;-$&$0$&$\;\;\;+$\\
\end{table} } 

  \end{incremental}
