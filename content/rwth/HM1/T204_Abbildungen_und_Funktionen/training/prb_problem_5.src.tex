	\begin{question}
    \explanation{Lösen Sie die Gleichung $y=\var{f}$ nach $x$ auf.}
		\begin{variables}
			\randint[Z]{a}{-10}{10}
			\randint{b}{-10}{10}
			\function[calculate]{c}{a/2}
			\function[calculate]{d}{b/2}
			\function[expand,normalize]{f}{c*x+d}
			\function{l}{(y-d)/c}
		\end{variables}
		\type{input.function}
		\field{rational}
		\text{Bestimmen Sie auf einem geeigneten Bereich die Umkehrfunktion der Funktion $f: \mathbb{R} \to \mathbb{R}, x \mapsto \var{f}$ und geben Sie die Funktionsvorschrift in der Variablen $y$ ein.}
		\begin{answer}
			\text{$f^{-1}(y)=$}
			\solution{l}
			\checkAsFunction{y}{-10}{10}{100}
		\end{answer}
	\end{question}

    \begin{question}
    \explanation{Lösen Sie die Gleichung $y=\var{f}$ nach $x$ auf.}
        \begin{variables}
			\randint[Z]{a}{-10}{10}
			\randint{b}{-10}{10}
			\function[calculate]{c}{a/2}
			\function[calculate]{d}{b/2}
			\randint{z}{-5}{5}
			\function[normalize]{f}{d+c/(x-z)}
			\function{l}{c/(y-d)+z}
		\end{variables}
		\type{input.function}
		\field{rational}
		\text{Bestimmen Sie auf einem geeigneten Bereich die Umkehrfunktion der Funktion $f: \mathbb{R} \backslash \{ \var{z} \} \to \mathbb{R}, x \mapsto \var{f}$ und geben Sie die Funktionsvorschrift in der Variablen $y$ ein.}
		\begin{answer}
			\text{$f^{-1}(y)=$}
			\solution{l}
			\checkAsFunction{y}{-10}{10}{100}
		\end{answer}
	\end{question}
\end{problem}
