\randomquestionpool{1}{5}    
\begin{question}

	\begin{variables}
		\randint[Z]{a}{-9}{9}
		\randint{b}{-9}{9}
		\randint[Z]{c}{-9}{9}
		\randint{d}{-9}{9}
		
		
			\matrix[calculate]{aa}{
  			a & b \\ 
  			c & d
      	}
      	
		\matrix[calculate]{m}{
		a & c \\
		b & d
      } 
	\end{variables}

	\type{input.matrix}
    \field{rational}
    
    \lang{de}{
	    \text{
	    Bestimmen Sie die Transponierte der folgenden Matrix:\\
$\var{aa}$}}
    
    \begin{answer}
	    \solution{m}
        \explanation{
            Für die Bestimmung der Transponierten einer Matrix,
            wird die Rolle der Zeilen und Spalten vertauscht.
        }
	\end{answer}
    
\end{question}
