\begin{block}[annotation]
  Rechnen mit Matrizen
     
\end{block}


\lang{de}{ Bestimmen Sie die transponierte Matrix zu:
\begin{enumerate}
 \item[(a)]
\[
 \begin{pmatrix}
  1 & -2 \\
  0 & 3
 \end{pmatrix}
\]
 \item[(b)]
\[
 \begin{pmatrix}
  1 & -2 & -5\\
  0 & 3 & 1\\
  0 & 4 & 1
 \end{pmatrix}
\]
 \item[(c)] 
\[
 \begin{pmatrix}
  2 & -5 & 6\\
  1 & 2 & 7
 \end{pmatrix}
\]
 \item[(d)]
\[
 \begin{pmatrix}
  1\\
  3 \\
  2
 \end{pmatrix}
\]
\end{enumerate}}


\lang{de}{ Bestimmen Sie:
\begin{enumerate}

\item[(e)]
\[
 \begin{pmatrix}
  1 & -2 \\
  0 & 3
 \end{pmatrix}^T
\cdot 
 \begin{pmatrix}
  1 & -2 \\
 \end{pmatrix}^T
\]

\item[(f)]
\[
 \left(
 \begin{pmatrix}
  1  \\
  3
 \end{pmatrix}
 +
 \begin{pmatrix}
  -2 \\
  4
 \end{pmatrix}
 \right)^T
 - 
 \begin{pmatrix}
  1 \\
  5
 \end{pmatrix}
\]

\end{enumerate}}


\begin{tabs*}[\initialtab{0}\class{exercise}]
  \tab{\lang{de}{Lösung (a)}}
  \lang{de}{
\[
 \begin{pmatrix}
  1 & -2 \\
  0 & 3
 \end{pmatrix}^T
=
 \begin{pmatrix}
  1 & 0 \\
  -2 & 3
 \end{pmatrix}
\]
}

 \tab{\lang{de}{Lösung (b)}}
  \lang{de}{\[
 \begin{pmatrix}
  1 & -2 & -5\\
  0 & 3 & 1\\
  0 & 4 & 1
 \end{pmatrix}^T
=
 \begin{pmatrix}
  1 & 0 & 0\\
  -2 & 3 & 4\\
  -5 & 1 & 1
 \end{pmatrix}
\]}

 \tab{\lang{de}{Lösung (c)}}
  \lang{de}{\[
 \begin{pmatrix}
  2 & -5 & 6\\
  1 & 2 & 7
 \end{pmatrix}^T
=
 \begin{pmatrix}
  2 & 1\\
  -5 & 2\\
  6 & 7
 \end{pmatrix}
\]}

 \tab{\lang{de}{Lösung (d)}}
  \lang{de}{\[
 \begin{pmatrix}
  1\\
  3 \\
  2
 \end{pmatrix}^T
=
\begin{pmatrix}
  1 & 3 & 2
 \end{pmatrix}
\]}

 \tab{\lang{de}{Lösung (e)}}
  \lang{de}{\[
 \begin{pmatrix}
  1 & -2 \\
  0 & 3
 \end{pmatrix}^T
\cdot 
 \begin{pmatrix}
  1 & -2 \\
 \end{pmatrix}^T
 = 
 \begin{pmatrix}
  1 & 0 \\
  -2 & 3
 \end{pmatrix}
\cdot 
 \begin{pmatrix}
  1 \\
  -2 \\
 \end{pmatrix}
 =
 \begin{pmatrix}
  1 \cdot 1 + 0 \cdot (-2) \\
  (-2) \cdot 1 + 3 \cdot (-2) \\
 \end{pmatrix}
 =
 \begin{pmatrix}
  1 \\
  -8 \\
 \end{pmatrix}
\]}

 \tab{\lang{de}{Lösung (f)}}
  \lang{de}{\[
 \left(
 \begin{pmatrix}
  1  \\
  3
 \end{pmatrix}
 +
 \begin{pmatrix}
  -2 \\
  4
 \end{pmatrix}
 \right)^T
 - 
 \begin{pmatrix}
  1 \\
  5
 \end{pmatrix}
 =
 \begin{pmatrix}
  -1  \\
  7
 \end{pmatrix}^T
 - 
 \begin{pmatrix}
  1 \\
  5
 \end{pmatrix}
 =
 \begin{pmatrix}
  -1 & 7  \\
 \end{pmatrix}
 - 
 \begin{pmatrix}
  1 \\
  5
 \end{pmatrix}
\]
Die Subtraktion kann nicht durchgeführt werden, da die Anzahl an Zeilen und Spalten nicht übereinstimmt.
Die Aufgabe ist nicht lösbar.
}

\end{tabs*}
