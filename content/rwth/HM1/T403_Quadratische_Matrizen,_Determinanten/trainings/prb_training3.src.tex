\begin{problem}
	%\randomquestionpool{}{}
	\begin{question}
		
		\begin{variables}
%a, b, c, d, f , g, h, i ∈ {− 4; . . . ; 4 } \ { 0 }


		\randint[Z]{a}{-4}{4} 
		\randint[Z]{b}{-4}{4} 
		\randint[Z]{c}{-4}{4} 
		\randint[Z]{d}{-4}{4} 
		\randint[Z]{f}{-4}{4} 
		\randint[Z]{g}{-4}{4} 
		\randint[Z]{h}{-4}{4} 		
		\randint[Z]{i}{-4}{4} 
		
\matrix[calculate]{aa}{
(a*f+1)*(1+c*h + d*i) + a*(g*h + i) &  a*(c*f + g) + c &  a*d*f + a + d\\
b*(c*f*h + d*f*i + f + g*h + i) + h &  b*(c*f + g) + 1 &  b*d*f + b\\
c*f*h + d*f*i + f + g*h + i &  c*f + g &  d*f + 1
}
\matrix[calculate]{bb1}{ 
d*f + 1 &  -c + d*g  & -a*d*f - a + b*c - b*d*g - d\\
-d*f*h - h &  (c - d*g)*h + 1 &  (a*d*f + a - b*c + b*d*g + d)*h - b\\
-d*f*i - f - i &  (c - d*g)*i - g &  (a*d*f  + a + b*(d*g -c) + d)*i + a*f + b*g + 1
} 
\matrix[calculate]{bb2}{ a*i + 1  & -a*c*i - c &  -a*d*i - a - d\\
b*i - h &  -b*c*i + c*h + 1 &  -b*d*i - b + d*h\\
g*(h - b*i) -a*f*i - f  - i &  c*(a*f*i + g*(b*i -h) + f + i) - g &  
d*(a*f*i+ b*g*i  + f -   g*h + i)+ b*g + a*f  + 1
} 
\matrix[calculate]{bb3}{ a*f + 1 &   a*(g-c*f) - c &  -a*d*f - a - d\\
(b -a*h)*f  - h &  a*(c*f - g)*h + b*(g -c*f) + c*h + 1 &  (a*h - b)*d*f + (a+ d)*h - b\\
-a*f*i - f - i &  c*(a*f*i+ f + i) - a*g*i  - g &  d*(a*f*i + f + i) + a*i + 1
}
			
			
		\end{variables}
		
		\type{input.generic}
        \field{real}
		\correctorprecision{3}
		\displayprecision{3}
		\lang{de}{
	    \text{Welche der folgenden Matrizen ist die inverse Matrix zu $A=\var{aa}$?}
	    }
     \lang{en}{
	    \text{Which of the following matrices is the inverse matric of $A=\var{aa}$?}
	    }
	    %\permuteAnswers{1, 2, 3} 
	    %http://team.mumie.net/projects/support/wiki/DifferentAnswerType
	    \lang{de}{
     \begin{answer}
            \type{mc.unique}
            \permutechoices{1}{3}
			\begin{choice}
			\text{$\var{bb1}$}
			\solution{true} 
			\end{choice}
			\begin{choice}
			\text{ $\var{bb2}$}
			\solution{false} %
			\end{choice}
			\begin{choice}
			\text{ $\var{bb3}$}
			\solution{false} %
			\end{choice}
            \explanation{Damit eine gegebene Matrix $B$ die Inverse zu $A$ ist, muss gelten $A\cdot B= E_3$. 
            Das Ausschlussprinzip ist hier zielführend.}
	    \end{answer}} 


     \lang{en}{
     \begin{answer}
            \type{mc.unique}
            \permutechoices{1}{3}
			\begin{choice}
			\text{$\var{bb1}$}
			\solution{true} 
			\end{choice}
			\begin{choice}
			\text{ $\var{bb2}$}
			\solution{false} %
			\end{choice}
			\begin{choice}
			\text{ $\var{bb3}$}
			\solution{false} %
			\end{choice}
            \explanation{For the inverse matrix $B$ of $A$ must hold $A\cdot B= I_3$. 
            Use the exclusion principle to find the inverse matrix.}
	    \end{answer}}
	    
	    %generic viz: http://team.mumie.net/projects/support/wiki/Example
	    
	\end{question}
\end{problem}
