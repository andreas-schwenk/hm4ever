\lang{de}{
a) Berechnen Sie in Abhängigkeit vom Parameter $c$ die Determinante zu
\begin{align*}
A=
\left( \begin{smallmatrix}
1 & 3 & 0 \\
0 & 1 & 2 \\
-1 & -2 & c 
\end{smallmatrix} \right).
\end{align*}

b) Für welche Werte von $c$ ist $A$ invertierbar?}


\lang{en}{
a) Calculate in dependence of the parameter $c$ the determinant of
\begin{align*}
A=
\left( \begin{smallmatrix}
1 & 3 & 0 \\
0 & 1 & 2 \\
-1 & -2 & c 
\end{smallmatrix} \right).
\end{align*}

b) For which values of $c$ is $A$ invertible?}
