\lang{de}{ Bestimmen Sie:
\begin{enumerate}[(a)]
 \item (a)
\[
 \begin{pmatrix}
  1 & -2 \\
  0 & 3
 \end{pmatrix}^T
\]
 \item (b)
\[
 \begin{pmatrix}
  1 & -2 & -5\\
  0 & 3 & 1-5i\\
  0 & 4+2i & 1
 \end{pmatrix}^T
\]
 \item (c)
\[
 \begin{pmatrix}
  2 & -5 & 6\\
  1 & 2 & 7
 \end{pmatrix}^T
\]
 \item (d)
\[
 \begin{pmatrix}
  1\\
  3 \\
  2
 \end{pmatrix}^T
\]
 \item (e)
\[
 \begin{pmatrix}
  1 & 2 & 4\\
  3 & -1 & 5
 \end{pmatrix}^T
 + 2 \cdot
 \begin{pmatrix}
  1 & 4 \\
  3 & 2 \\
  -1 & 2
 \end{pmatrix}
\]
\end{enumerate}
}
