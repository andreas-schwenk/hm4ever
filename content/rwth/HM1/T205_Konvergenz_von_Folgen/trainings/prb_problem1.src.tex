     \begin{question} %fall q zwischen 0 und 1
     \lang{de}{ 
      	\text{Untersuchen Sie die Folge $(a_{n})_{n\in\N}$ 
      	mit $a_{n}=\var{u} \cdot (\var{q})^{n}$ für alle $n\in\N$ auf Monotonie und Beschränktheit.
		}
    	}
    	\type{mc.yesno}
    	\precision[false]{3}
    	\displayprecision{3}
    
    	\begin{variables}
    	\randint[Z]{u}{-10}{10}
    	%\randint[Z]{q}{-5}{5}
    	
    	\randint{qqnenner}{6}{10}
    	\randint{qqzahler}{1}{5}
    	\function{q}{qqzahler/qqnenner}
    	
    	
    	\end{variables}
		\begin{choice}
			\text{$(a_{n})_{n\in\N}$ ist monoton fallend.}
			\solution{compute}
			\iscorrect{u}{>}{0}
		\end{choice}
		\begin{choice}
			\text{$(a_{n})_{n\in\N}$ ist monoton steigend.}
			\solution{compute}
			\iscorrect{u}{<}{0}
		\end{choice}
		\begin{choice}
			\text{$(a_{n})_{n\in\N}$ ist beschränkt}
			\solution{compute}
			\iscorrect{abs(q)}{<=}{1}
		\end{choice}
    \end{question}
