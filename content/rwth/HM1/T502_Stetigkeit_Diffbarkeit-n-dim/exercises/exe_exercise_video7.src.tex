% Aufgabe 9

Berechnen Sie zu den folgenden Funktionen sämtliche partielle Ableitungen erster und zweiter Ordnung.

\begin{itemize}

\item $f:\R^3 \rightarrow\R$, $f(x,y,z)= \frac{x^2 y}{z^2 +1}$. 

\item  $f:\R^2 \rightarrow\R$, $f(x,y)=e^{xy}\sin(x^2 +y)$.

\end{itemize}

\tab{\lang{de}{Antworten}}
  \begin{enumerate}[alph]
      \item $\frac{\partial }{\partial x}\frac{\partial}{\partial z}f= 
      \frac{\partial}{\partial z}\frac{\partial }{\partial x}f= 
      \frac{-4xyz}{(z^2 +1)^2}$. 

      \item $\frac{\partial }{\partial x}\frac{\partial}{\partial y}f =
      \frac{\partial}{\partial y}\frac{\partial }{\partial x}f= $\\
      $e^{xy}(xy\sin(x^2+y)+\sin(x^2+y)+y\cos(x^2+y)+2x^2 \cos(x^2+y)-2x\sin(x^2+y))$.
      
  \end{enumerate}
