			\randint{a}{-9}{9}

			\randint{ee}{1}{9}
			\randint{f}{1}{9}
			\randint{g}{1}{9}
			
			\function{an}{(a*n^2+b*n+c)/(d*n^3+ee*n^2+f*n+g)}
			
		\end{variables}
	
		\lang{de}{ 
      		\text{Untersuchen Sie folgende Folgen auf Konvergenz, bestimmte Divergenz und unbestimmte Divergenz:\\ \\$(a_n)_{n\in\N}$ mit $a_n=\var{an}$}
    		\explanation{Betrachten Sie die Terme mit den höchsten Exponenten im Zähler und Nenner sowie deren Vorzeichen.}
    	}
    	\type{mc.unique} 
    	
    	\begin{choice}
    	 	\text{ Konvergenz }
  		 	\solution{true}
		\end{choice}
    	\begin{choice}
    	 	\text{ Bestimmte Divergenz gegen $\infty$ }
  		 	\solution{false}
		\end{choice}

    	 	\text{ Bestimmte Divergenz gegen $-\infty$ }
  		 	\solution{false}
		\end{choice}
    	\begin{choice}
    	 	\text{ Unbestimmte Divergenz }
  		 	\solution{false}
		\end{choice}
		
    \end{question}
