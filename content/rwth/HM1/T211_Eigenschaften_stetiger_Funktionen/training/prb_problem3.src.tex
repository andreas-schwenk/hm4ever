\begin{problem}
    
    \begin{question} 
     
     	
      	\text{Gegeben sei die Funktion $f:\R\to\R, \ f(x)=\cos(x)+c$. 
        Bestimmen Sie die Menge aller $c$, 
        für die $f$ unendlich viele Nullstellen besitzt.}
    	\type{mc.unique}
	\permutechoices{1}{7}
		\begin{choice}
			\text{$\{c \ | \  c \in \mathbb{R}\}$}
			\solution{false}
		\end{choice}
		\begin{choice}
			\text{$\{c \ | \ |c| \le 1\}$}
			\solution{true}
		\end{choice}
		\begin{choice}
			\text{$\{c \ | \ c < 0\}$}
			\solution{false}
		\end{choice}
		\begin{choice}
			\text{$\{c \ | \ c > 0\}$}
			\solution{false}
		\end{choice}
		\begin{choice}
			\text{$\{c \ | \ c \ge 1\}$}
			\solution{false}
		\end{choice}
		\begin{choice}
			\text{$\{c \ | \ c \le 1\}$}
			\solution{false}
		\end{choice}
		\begin{choice}
			\text{$\{c \ | \ c \ge -1\}$}
			\solution{false}
		\end{choice}
        \explanation[edited]{Hier muss nur berücksichtigt werden, dass
        $\cos(x)\in [-1;1]$ und Kosinus periodisch ist.}
	\end{question}

\end{problem}
