\begin{question}

	\begin{variables}
		\randint{a}{-3}{3}
        \randint{b}{-3}{3}
        \randint{g}{-3}{3}
        \randint{f}{-3}{3}
        \randint{c}{-5}{5}
        \randint{d}{-5}{5}
        
        \pmatrix{aa}{a & b \\ g & f  }
        
        \pmatrix[calculate]{dd}{1 \\ d+1}
        \pmatrix[calculate]{cc}{c+1 \\ c*d+c+d}
		\pmatrix[calculate]{bb}{(d+1)*a-g & (d+1)*b - f \\
		(-c*d-c-d)*a + (c+1)*g & (-c*d-c-d)*b+(c+1)*f	}
        \pmatrix{dumm}{x \\ y}
	\end{variables}

	\type{input.matrix}
	\displayprecision{3}
    \correctorprecision{2}
    \field{real}
    
    \lang{de}{
	    \text{ Bestimmen Sie die Abbildungsmatrix ${\,}_B f_S$, wobei $S$ die Standardbasis des $\R^2$ ist, 
        $B=\left\{\var{cc},\var{dd} \right\}$ und \\ $f : \R^2 \to \R^2, \var{dumm} \mapsto \var{aa}\cdot \var{dumm}$
      }
      \explanation{ Die Abbildungsmatrix ${\,}_B f_S = \begin{pmatrix} a_11 & a_12 \\ a_21 & a_22\end{pmatrix}$
      ist eine $2\times 2-$Matrix und hat als Spalten die beiden Koordinatenvektoren von 
      ${\~}f\begin{pmatrix}1 \\ 0\end{pmatrix}=\begin{pmatrix} \var{a} \\ \var{c} \end{pmatrix}$ und 
      ${\~}f\begin{pmatrix}0 \\ 1\end{pmatrix}=\begin{pmatrix} \var{b} \\ \var{d} \end{pmatrix}$ bezüglich 
      der Basis $B$. 
      \\
      Es müssen also die folgenden linearen Gleichungssysteme gelöst werden
     \\
      $\begin{pmatrix} \var{a} \\ \var{c} \end{pmatrix} = a_11 \cdot \var{cc}  + a_21 \cdot \var{dd}$
%       für die 1.Spalte und 
       und
%       \\
      $\begin{pmatrix} \var{b} \\ \var{d} \end{pmatrix} = a_12 \cdot \var{cc}  + a_22 \cdot \var{dd}$.
%        für die 2.Spalte von ${\,}_B f_S = \begin{pmatrix} a_11 & a_12 \\ a_21 & a_22\end{pmatrix}$. 
      }  
      }


         \lang{en}{
	    \text{ Determine the matrix ${\,}_B f_S$ associated with a linear map, at which $S$ is the standard basis of $\R^2$, 
        $B=\left\{\var{cc},\var{dd} \right\}$ and \\ $f : \R^2 \to \R^2, \var{dumm} \mapsto \var{aa}\cdot \var{dumm}$
      }
      \explanation{ The matrix of the linear map ${\,}_B f_S = \begin{pmatrix} a_11 & a_12 \\ a_21 & a_22\end{pmatrix}$
      is a $2\times 2-$matrix and the columns are the coordinate vectors of 
      ${\~}f\begin{pmatrix}1 \\ 0\end{pmatrix}=\begin{pmatrix} \var{a} \\ \var{c} \end{pmatrix}$ and 
      ${\~}f\begin{pmatrix}0 \\ 1\end{pmatrix}=\begin{pmatrix} \var{b} \\ \var{d} \end{pmatrix}$ with respect
      to the basis $B$. 
      \\
      So we need to solve the following system of linear equations
     \\
      $\begin{pmatrix} \var{a} \\ \var{c} \end{pmatrix} = a_11 \cdot \var{cc}  + a_21 \cdot \var{dd}$
%       für die 1.Spalte und 
       and
%       \\
      $\begin{pmatrix} \var{b} \\ \var{d} \end{pmatrix} = a_12 \cdot \var{cc}  + a_22 \cdot \var{dd}$.
%        für die 2.Spalte von ${\,}_B f_S = \begin{pmatrix} a_11 & a_12 \\ a_21 & a_22\end{pmatrix}$. 
      }  
      }
    
    \begin{answer}
	    \solution{bb}
      %\format{2}{2}
	\end{answer}
    
\end{question}
