\usepackage{mumie.genericproblem}

     \begin{problem}
     \begin{variables}
            \randint{n}{5}{7}
            \randint{r}{45}{55}
            \randint{p}{1}{4}
            \function[calculate,2]{rsub}{r*(12+p*13/200)}
            \function[calculate,2]{q}{1+p/100}
            \function[calculate,2]{rm}{rsub*(q^n-1)/(q-1)}
           
     \end{variables}
          \begin{question}
          \type{input.number}
          \lang{de}{\text{Für einen vermögenswirksamen Sparvertrag werden $\var{n}$ Jahre lang jeweils am
          Monatsanfang \var{r} \euro eingezahlt. Der Zins beträgt während der gesamten Laufzeit $\var{p}$ \%.
          Wie hoch ist die jährliche Ersatzzahlung und wie lautet der Kontostand $R_n$ nach $\var{n}$ Jahren?\\
          Jährliche Ersatzzahlung \ansref \euro\\
          Rentenendwert:\ansref \euro\\
          }}
          \lang{en}{\text{For a capital-forming savings contract, $\var{n}$ is paid out \euro at the beginning of each month. 
          The interest is $\var{p}$ \% over the entire term.
          What is the annual replacement payment and what is the account balance $R_n$ after $\var{n}$ years?\\
          Annual replacement payment \ansref \euro\\
          Final future value of the annuity:\ansref \euro\\
          }}
               \begin{answer}
                    \solution{rsub}
               \end{answer}
               \begin{answer}
                    \solution{rm}
               \end{answer}               
               \lang{de}{\explanation[edited]{Die monatliche Zahlung ist vorschüssig, so dass sich für die 
               jährliche Ersatzzahlung ergibt:
               $r_e=\var{r}\cdot(12+\frac{13\cdot\var{p}}{200})=\var{rsub}$ \euro. \\
               ($r_e$ ergibt sich aus den eingezahlten monatlichen Raten mit einfacher Verzinsung).
               %Damit ergibt sich mit dem Zinsfaktor $q=\var{q}$ der 
               %Rentenendwert zu: $R_n=r_e\frac{q^n-1}{q-1}=\var{rm}$ \euro.
               }}
               \lang{en}{\explanation[edited]{The monthly payment is in advance, so that for the 
               annual replacement payment:
               $r_e=\var{r}\cdot(12+\frac{13\cdot\var{p}}{200})=\var{rsub}$ \euro. \\
               ($r_e$ results from the paid-in monthly instalments with simple interest).
               %With the interest factor $q=\var{q}$ this gives the 
               %final future value to: $R_n=r_e\frac{q^n-1}{q-1}=\var{rm}$ \euro.
               }}
          \end{question}     
     \end{problem}
