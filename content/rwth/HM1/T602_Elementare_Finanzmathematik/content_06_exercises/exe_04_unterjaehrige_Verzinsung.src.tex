\title{\lang{en}{Ü04: effective, nominal and conformal interest rate}
    \lang{de}{Ü04: effektiver, nomineller und konformer Zinssatz}
    \lang{zh}{...}
    \lang{fr}{...}}

\lang{de}{Wir betrachten ein Kapital von $K_0=2000$ \euro auf einem Girokonto mit einen relativen Periodenzinssatz
von $i_m=2$ \% je Quartal. Der Beobachtungszeitraum sei ein Jahr. }
\lang{en}{We observe a capital of $K_0=2000$ \euro in a checking account with a relative 
quarterly interest rate of $i_m=2$ over a period of one year.
}

\begin{itemize}
    \item[a)] \lang{de}{Berechnen Sie das Kapital bei \textbf{unterjähriger Verzinsung} nach einem Jahr.} \lang{en}{Calculate the value with \textbf{quarterly compounding interest} after one year.}
    \item[b)] \lang{de}{Berechnen Sie den nominellen Jahreszins $i_{\text{nom}}$.} \lang{en}{Calculate the nominal annual interest rate $i_{\text{nom}}$.}
    \item[c)] \lang{de}{Berechnen Sie den effektiven Jahreszins $i_{\text{eff}}$.} \lang{en}{Calculate the effective annual interest rate $i_{\text{eff}}$.}
    \item[d)] \lang{de}{Berechnen Sie den konformen Zinssatz $i'$.} \lang{en}{Calculate the conformal interest rate $i'$.}
\end{itemize}

    \begin{incremental}[\initialsteps{1}]
        \step \lang{de}{Die Lösung ist:
        \begin{align*}
        K_{1;4}=2164,86 \text{ \euro}
        \end{align*}}
        \lang{en}{The solution is:
        \begin{align*}
        K_{1;4}=2164.86 \text{ \euro}
        \end{align*}}
        \step
\[K_{n;k}=K_0\left(1+i_m\right)^{mn}\]
\lang{de}{Mit $m=4$ und $n=1$ gilt:} \lang{en}{Setting $m=4$ and $n=1$, we find:}
\lang{de}{\[K_{1;4}=2000(1+0,02)^4=2000\cdot1,0824321=2164,86 \text{ \euro}\]}
\lang{en}{\[K_{1;4}=2000(1+0.02)^4=2000\cdot1.0824321=2164.86 \text{ \euro}\]}
    \end{incremental}
\tab{\lang{de}{Lösung} \lang{en}{Solution} b)}   
    \begin{incremental}[\initialsteps{1}]
    \step 
    \lang{de}{Die Lösung ist:} \lang{en}{The solution is:} 
    \begin{align*}
        i_{\text{nom}}=8\%
    \end{align*}
    \step
    \lang{de}{Der nominelle Jahreszins ermittelt sich ganz einfach aus $i_{\text{nom}}=m\cdot i_m$:}
    \lang{en}{The nominal annual interest is simply $i_{\text{nom}}=m\cdot i_m$:}
    \[i_{\text{nom}}=4\cdot 2\%=8\%\]
    \end{incremental}
\tab{\lang{de}{Lösung} \lang{en}{Solution} c)}
    \begin{incremental}[\initialsteps{1}]
        \step \lang{de}{Die Lösung ist:
        \begin{align*}
        i_{\text{eff}}=8,24 \%
        \end{align*}}
        \lang{en}{The solution is:
        \begin{align*}
        i_{\text{eff}}=8.24 \%
        \end{align*}}
        \step    
        \lang{de}{Der effektive Jahreszins ist der Zinssatz, der bei jährlicher Verzinsung zum selben Ergebnis führt 
        wie die unterjährige Verzinsung:\\
        \[2000\cdot\left(1+\frac{p_{\text{eff}}}{100}\right)^1=2164,86\]
        Somit ist $i_{\text{eff}}=8,24\%$.}
        \lang{en}{The effective annual interest rate is the rate at which annually compounded interest would yield the
        same result as quarterly compounded interest at the rate $i_m$:
        \[2000\cdot\left(1+\frac{p_{\text{eff}}}{100}\right)^1=2164.86\]
        Therefore, $i_{\text{eff}}=8.24\%$.}
     \end{incremental}
\tab{\lang{de}{Lösung} \lang{en}{Solution} d)}
    \begin{incremental}[\initialsteps{1}]
        \step\lang{de}{ Die Lösung ist:
        \begin{align*}
        i'=1,94 \%
        \end{align*}}
        \lang{en}{The solution is:
        \begin{align*}
        i'=1.94 \%
        \end{align*}}
        \step
        \lang{de}{Der konforme Zinssatz ist der periodische Zinssatz, der zum gleichen Ergebnis führt wie der
        der nominelle Zinssatz:
        \[2000\left(1+\frac{p'}{100}\right)^4=2000\cdot\left(1+\frac{8}{100}\right)^1=2000\cdot 1,08=2160 \]
        Dies formen wir um zu: 
        \[i'=\sqrt[4]{\frac{2160}{2000}}-1=1,94\%\]}
        \lang{en}{The conformal interest rate is the periodic interest rate that produces the same result as the nominal interest rate:
        \[2000\left(1+\frac{p'}{100}\right)^4=2000\cdot\left(1+\frac{8}{100}\right)^1=2000\cdot 1.08=2160 \]
        We can rewrite this equation in the form 
        \[i'=\sqrt[4]{\frac{2160}{2000}}-1=1.94\%\]}
        \end{incremental}
\end{tabs*}
