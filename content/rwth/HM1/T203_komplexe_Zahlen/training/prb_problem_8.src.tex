\usepackage{mumie.genericproblem}
\title{A08: neutrales/inverses Element}

     \begin{problem}
     
     \begin{variables}
        \randint{a}{-6}{6}
        \randint[Z]{b}{-6}{6}
        \function[expand, normalize]{z}{a+b*i}
        \function[expand, normalize]{za}{-a-b*i}
        \function[calculate,expand, normalize]{zm}{(a-b*i)/(a^2+b^2)}
        \function[expand, normalize]{zq}{a-b*i}   
        \function[expand, normalize]{zb}{a^2+b^2}
        \function{z0}{0}
        \function{z1}{1}
     \end{variables}
          \begin{question}
               \field{complex}
               \text{Bestimmen Sie das \textcolor{#CC6600}{neutrale und inverse Element bzgl. 
                der Addition} von $z=\var{z}$.}
               \type{input.function}
               \begin{answer}
                    \text{$\textcolor{#CC6600}{-z}=$}
                    \solution{za}
                    \inputAsFunction{x}{antwort}
                    \checkFuncForZero{za - antwort}{1}{2}{1}
               \end{answer}
               \begin{answer}
                    \text{$\textcolor{#CC6600}{z_0^+}=$}
                    \solution{z0}
                    \checkAsFunction{x}{1}{2}{1}
               \end{answer}
               \explanation{Für $z=a+bi$ ist $-z=-(a+bi)=-a-bi=\var{za}$.}
          \end{question} 
          \begin{question}
               \field{complex}
               \type{input.function}
               \text{Bestimmen Sie das \textcolor{#0066CC}{neutrale und inverse Element bzgl. 
                der Multiplikation} von $z=\var{z}$.\\
                (Geben Sie $z^{-1}$ bitte in der Form $z=a+bi$ ein.)
                }
               \begin{answer}
                    \text{$\textcolor{#0066CC}{z^{-1}}=$}
                    \solution{zm}
                    \inputAsFunction{x}{antwort}
                    \checkFuncForZero{zm - antwort}{1}{2}{1}
               \end{answer} 
               \begin{answer}
                    \text{$\textcolor{#0066CC}{z_1^*}=$}
                    \solution{z1}
                    \checkAsFunction{x}{1}{2}{1}
               \end{answer}               
               \explanation{Für $z=a+bi$ ist  $z^{-1}=\frac{\bar{z}}{z\cdot\bar{z}}=\frac{a-bi}{a^2+b^2}
               =\frac{\var{zq}}{\var{zb}}.$
            }
          \end{question}          
     \end{problem}
