\lang{de}{ 
Bestimmen Sie die Länge folgender Vektoren.
\begin{enumerate}[a)]
 \item a)
  \[
   \begin{pmatrix}1\\ 7 \end{pmatrix}
  \]
 \item b)
  \[
   \begin{pmatrix}2\\ -3 \end{pmatrix}
  \]
 \item c)
  \[
   \begin{pmatrix}4\\ -6 \\ -9 \end{pmatrix}
  \]
 \item d)
  \[
   \begin{pmatrix}\frac{1}{\sqrt{3}}\\ \frac{1}{\sqrt{3}}\\ \frac{1}{\sqrt{3}}  \end{pmatrix}
  \]
\end{enumerate}}

\begin{table}[\class{items}]
a) $\sqrt{50}$\\
b) $\sqrt{13}$\\
c) $\sqrt{133}$\\
d) $1$
\end{table}
  \tab{
  \lang{de}{Lösung a)}}
  \[
   \norm{\begin{pmatrix}1\\ 7 \end{pmatrix}}=\sqrt{1^2+7^2}=\sqrt{50}
  \]

 \tab{\lang{de}{Lösung b)}}
 \[
   \norm{\begin{pmatrix}2\\ -3 \end{pmatrix}}=\sqrt{2^2+(-3)^2}=\sqrt{4+9}=\sqrt{13}
  \]
	
	 \tab{\lang{de}{Lösung c)}}
	 \[
   \norm{\begin{pmatrix}4\\ -6\\ -9 \end{pmatrix}}=\sqrt{4^2+(-6)^2+(-9)^2}=\sqrt{16+36+81}=\sqrt{133}
  \]
	
	 \tab{\lang{de}{Lösung d)}}
	 \[
   \norm{\begin{pmatrix}\frac{1}{\sqrt{3}}\\ \frac{1}{\sqrt{3}}\\ \frac{1}{\sqrt{3}}  \end{pmatrix}}=\sqrt{\left(\frac{1}{\sqrt{3}}\right)^2+\left(\frac{1}{\sqrt{3}}\right)^2+\left(\frac{1}{\sqrt{3}}\right)^2}=\sqrt{\frac{1}{3}+\frac{1}{3}+\frac{1}{3}}=1
  \]



\end{tabs*}
