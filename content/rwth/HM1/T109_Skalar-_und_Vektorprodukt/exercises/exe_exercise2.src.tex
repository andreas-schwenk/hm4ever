\lang{de}{ 
Bestimmen Sie folgende Skalarprodukte.
\begin{enumerate}[a)]
 \item a)
\[
 \ \begin{pmatrix}1\\ 4\\ -5 \end{pmatrix} \bullet \begin{pmatrix}2\\ -3\\ 6 \end{pmatrix} \,
\]
 \item b)
\[
 \ \begin{pmatrix}1\\ 3\\ -2 \end{pmatrix}\bullet \begin{pmatrix}3\\ 0\\ 0 \end{pmatrix} \,
\]
 \item c)
\[
 \ 2\cdot\begin{pmatrix}1\\ 2 \\ 6 \end{pmatrix}\bullet  \begin{pmatrix}2\\ -4 \\ 2 \end{pmatrix} \,
\]
 \item d)
\[
 \ \left(\begin{pmatrix}5\\ 3\\ 4 \end{pmatrix}+\begin{pmatrix}-2\\ -9\\ 6 \end{pmatrix}\right)\bullet \left(\begin{pmatrix}4\\ 2 \\ 8 \end{pmatrix}-\begin{pmatrix}-1\\ -3\\ 5 \end{pmatrix}\right) \,
\]
\end{enumerate}}

  \lang{de}{Lösung a)}}



  \lang{de}{\[
   \  \begin{pmatrix}1\\ 4\\ -5 \end{pmatrix} \bullet \begin{pmatrix}2\\ -3\\ 6 \end{pmatrix} \,=1\cdot 2+4\cdot (-3)+(-5)\cdot 6=2-12-30=-40
	\]}

 \tab{\lang{de}{Lösung b)}}
  \lang{de}{\[
  \ \begin{pmatrix}1\\ 3\\ -2 \end{pmatrix}\bullet \begin{pmatrix}3\\ 0\\ 0 \end{pmatrix}\,=1\cdot 3+3\cdot 0+(-2)\cdot 0=3
	\]}
	
	 \tab{\lang{de}{Lösung c)}}
  \lang{de}{\[
  \ 2\cdot\begin{pmatrix}1\\ 2 \\ 6 \end{pmatrix}\bullet \begin{pmatrix}2\\ -4 \\ 2 \end{pmatrix}\, =\ \begin{pmatrix}2\\ 4\\ 12 \end{pmatrix} \bullet \begin{pmatrix}2\\ -4\\ 2 \end{pmatrix} \, =2\cdot 2+4\cdot (-4)+12\cdot 2=4-16+24=12
	\]}
	
	 \tab{\lang{de}{Lösung d)}}
  \lang{de}{\[
  \ \left(\begin{pmatrix} 5\\ 3\\ 4 \end{pmatrix}+\begin{pmatrix}-2\\ -9\\ 6 \end{pmatrix}\right)\bullet \left(\begin{pmatrix}4\\ 2 \\ 8 \end{pmatrix}-\begin{pmatrix}-1\\ -3\\ 5 \end{pmatrix}\right)\, =\  \begin{pmatrix}3\\ -6\\ 10 \end{pmatrix} \bullet \begin{pmatrix}5\\ 5\\ 3 \end{pmatrix}\, =3\cdot 5+(-6)\cdot 5+10\cdot 3=15-30+30=15
	\]}
