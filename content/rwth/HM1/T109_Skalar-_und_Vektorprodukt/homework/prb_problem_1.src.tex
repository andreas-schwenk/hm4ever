\begin{question} % question 1
      \lang{de}{\text{W\"{a}hlen Sie alle richtigen Aussagen aus:}}
      \lang{en}{\text{Select all the correct statements:} }
      \lang{de}{\explanation{Schauen Sie sich die Definition der Zahlenmengen nochmals an.}}
      \lang{en}{\explanation{Take another look at the definition of the number sets.}}

      \type{mc.multiple}
      \field{rational} 

      \begin{variables}
            \randint[Z]{a}{-10}{10}
            \randint{b}{-10}{10}
            \randint{c1}{-10}{10}
            \randint{c2}{2}{5}
            \function[calculate]{c}{c1*c2}
            \randint[Z]{d1}{-10}{10}
            \randint{d2}{1}{5}
            \randint{f}{0}{6}
      \end{variables}
      
     \permutechoices{1}{5}
      \begin{choice}
            \lang{de}{\text{$\var{a}$ ist eine nat"urliche Zahl.}}
            \lang{en}{\text{$\var{a}$ is a natural number.}}
            \solution{compute}
            \iscorrect{a}{>}{0}
      \end{choice}

      \begin{choice}
            \lang{de}{\text{$\var{b}$ ist eine rationale Zahl.}}
	    \lang{en}{\text{$\var{b}$ is a rational number.}}
           \solution{true}
      \end{choice}
      
            \begin{choice}
           \lang{de}{\text{$\frac{\var{c}}{\var{c2}}$ ist eine ganze Zahl.}}
           \lang{en}{\text{$\frac{\var{c}}{\var{c2}}$ is an integer.}}
           \solution{true}
      \end{choice}
      
      \begin{choice}
           \lang{de}{\text{$\var{d1}\cdot (-\var{d2})$ ist eine nat"urliche Zahl.}}
           \lang{en}{\text{$\var{d1}\cdot (-\var{d2})$ is a natural number.}}
            \solution{compute}
            \iscorrect{d1}{<}{0} 
      \end{choice}
      
      \begin{choice}
           \lang{de}{\text{$1:\var{f}$ ist eine ganze Zahl.}}
           \lang{en}{\text{$1:\var{f}$ is an integer.}}
            \solution{compute} %
            \iscorrect{f}{=}{1}
      \end{choice}
      
      \begin{choice}
      		\lang{de}{\text{Keine der Aussagen ist richtig.}}
      		\lang{en}{\text{None of the above statments is correct.}}
      		\solution{false}
      \end{choice}
\end{question}

\begin{question} % question 4
      \lang{de}{\text{Welche der folgenden Ausdr체cke haben als Ergebnis die Zahl $\var{erg}$?}}
      \lang{en}{\text{Which of the following expressions has the number $\var{erg}$ as the answer?}}
      \lang{de}{\explanation{Berechnen Sie die Ausdr체cke mit den Regeln: Klammer zuerst, Punkt-vor-Strich, von-links-nach-rechts.}}
      \lang{en}{\explanation{Calculate the expressions using \textbf{BEDMAS}.}}

      \type{mc.multiple}
      \field{rational} 

      \begin{variables}
            \randint{a}{2}{5}
             \function[calculate]{b}{a+1}
             \function[calculate]{c}{2*a}
             \function[calculate]{d}{a-1}
             \function[calculate]{s1}{a+b*c-d}
             \function[calculate]{s2}{(a+b)*(c-d)}
             \function[calculate]{s3}{a*(b+c)-d}
             \function[calculate]{s4}{c+a*(d-b)}
             \function[calculate]{s5}{(c/(b-d))*a}
             \randint{sel}{1}{5}
             \function[calculate]{erg}{s1*(1-(sign(sel-1))^2)+s2*(1-(sign(sel-2))^2)+s3*(1-(sign(sel-3))^2)+s4*(1-(sign(sel-4))^2)+s5*(1-(sign(sel-5))^2)}
      \end{variables}

      \permutechoices{1}{5}
      \begin{choice}
            \text{$\var{a}+\var{b}\cdot \var{c}-\var{d}$}
            \solution{compute}
            \iscorrect{s1}{=}{erg}
      \end{choice}

      \begin{choice}
            \text{$(\var{a}+\var{b})\cdot (\var{c}-\var{d})$}
            \solution{compute}
            \iscorrect{s2}{=}{erg}
      \end{choice}
      \begin{choice}
            \text{$\var{a}\cdot (\var{b}+ \var{c})-\var{d}$}
            \solution{compute}
            \iscorrect{s3}{=}{erg}
      \end{choice}
      \begin{choice}
            \text{$\var{c}+\var{a}\cdot (\var{d}-\var{b})$}
            \solution{compute}
            \iscorrect{s4}{=}{erg}
      \end{choice}
      \begin{choice}
            \lang{de}{\text{$\var{c}:(\var{b}-\var{d})\cdot \var{a}$}}
            \lang{en}{\text{$\var{c}/(\var{b}-\var{d})\cdot \var{a}$}}
            \solution{compute}
            \iscorrect{s5}{=}{erg}
      \end{choice}
      \begin{choice}
            \lang{de}{\text{Kein Ausdruck hat als Ergebnis die Zahl $\var{erg}$}}
            \lang{en}{\text{None of the expressions is equal to the number $\var{erg}$.}}
            \solution{false}
      \end{choice}

\end{question}
