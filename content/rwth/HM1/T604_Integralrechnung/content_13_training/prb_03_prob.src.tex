\usepackage{mumie.genericproblem}

     \begin{problem}
          \begin{question}
          \type{mc.yesno}
          \allMustBeCorrect
          \lang{de}{\text{Die Fläche unterhalb der Grenzumsatzkurve zwischen $x=0$ und 
          $x=x_0$ gibt bekanntlich den Gesamtumsatz an der Stelle $x=x_0$ an.
          
          \\Aussage:\\
          Die Fläche unterhalb der Grenzkostenkurve im gleichen Intervall 
          gibt aber nicht die Gesamtkosten bei $x=x_0$ an.\\
          Wählen Sie alle richtigen Antworten aus.
          }}
          \lang{en}{\text{The area below the marginal revenue curve between $x=0$ and 
          $x=x_0$ gives, as we know, the total turnover at the point $x=x_0$.
          
          \\Claim:\\
          The area under the marginal cost curve in the same interval 
          does not give the total cost at $x=x_0$.\\
          Select all the correct answers.
          }}
               \begin{choice}
                \lang{de}{\text{Die Aussage ist wahr.}}
                \lang{en}{\text{The claim is true.}}
                \solution{true}
               \end{choice}
               
               \begin{choice}
                \lang{de}{\text{Die Gesamtkosten an der Stelle $x=0$ sind Null.}}
                \lang{en}{\text{The total cost at the point $x=0$ is zero.}}
                \solution{false}
               \end{choice}
               
               \begin{choice}
                \lang{de}{\text{Bei $x=0$ müssen immer noch die vorhanden fixen Kosten
                mitberücksichtigt werden.}}
                \lang{en}{\text{At $x=0$, the existing fixed costs must still be taken into account.}}
                \solution{true}
               \end{choice}
               
               \begin{choice}
                \lang{de}{\text{Die fixen Kosten haben mit der Integrationskonstante nichts zu tun.}}
                \lang{en}{\text{The fixed costs have nothing to do with the integration constant.}}
                \solution{false}
               \end{choice}
               
               \lang{de}{\explanation{Bei $x=0$ wurde nichts verkauft, also kein Umsatz erzielt. Allerdings fallen
               bei $x=0$ trotzdem Kosten an, nämlichen die fixen Kosten. Diese fixen Kosten entsprechen genau der
               Integrationskonstanten.}}
               \lang{en}{\explanation{When $x=0$, nothing was sold, so no turnover was achieved. However,
               costs are still incurred at $x=0$, namely the fixed costs. These fixed costs correspond exactly to the
               integration constant}}
          \end{question}     
     \end{problem}
