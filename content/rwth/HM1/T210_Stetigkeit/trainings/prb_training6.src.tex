\lang{de}{\title{A06: Grenzwert $x\to\pm\infty$ }}
\begin{problem}
    \begin{question}
        \type{input.function}
        \field{real}
        \begin{variables}
            \randint{a}{5}{8}
            \function{n1}{-1/a}
            \function{n2}{0}
        \end{variables}        
        \lang{de}{\text{Berechnen Sie folgenden Grenzwert:}
        }
        \begin{answer}
            \text{$\lim_{x\to+\infty}\frac{e^{2x}-e^{5x}}{\var{a}e^{5x}+e^x}=$}
            \solution{n1}
            \checkAsFunction{a}{-10}{10}{100}
            \explanation{$\lim_{x\to+\infty}\frac{e^{2x}-e^{5x}}{\var{a}e^{5x}+e^x}=\lim_{x\to\infty}\frac{e^{5x}(e^{-3x}-1)}{e^{5x}(\var{a}+e^{-4x})}=
            \lim_{x\to+\infty}\frac{e^{-3x}-1}{\var{a}+e^{-4x}}=
            \frac{\lim_{x\to+\infty}e^{-3x}-1}{\lim_{x\to+\infty}\var{a}+e^{-4x}}=-\frac{1}{\var{a}}$}   
        \end{answer}
    \end{question}
     \begin{question}
        \type{input.function}
        \field{real}
        \begin{variables}
            \randint{a}{5}{8}
            \function{n1}{-1/a}
            \function{n2}{0*a}
        \end{variables}        
        \lang{de}{\text{Berechnen Sie folgenden Grenzwert:}
        }
        \begin{answer}
            \text{$\lim_{x\to-\infty}\frac{e^{2x}-e^{5x}}{\var{a}e^{5x}+e^x}=$}
            \solution{n2}
            \checkAsFunction{a}{-10}{10}{100}
            \explanation{$\lim_{x\to-\infty}\frac{e^{2x}-e^{5x}}{\var{a}e^{5x}+e^x}=
            \lim_{x\to-\infty}\frac{e^x(e^x-e^{4x})}{e^x(\var{a}e^{4x}+1)}=
            \frac{\lim_{x\to-\infty}e^x-e^{4x}}{\lim_{x\to-\infty}\var{a}e^{4x}+1}=\frac{0}{1}=0
            $}
        \end{answer}
    \end{question}
    
\end{problem}
