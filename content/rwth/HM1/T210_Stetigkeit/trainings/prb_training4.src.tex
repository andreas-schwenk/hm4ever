\begin{question}

	
\begin{variables}
	\randint[Z]{a}{-10}{10}
	\randint{b}{-10}{10}
	\randint{c}{-10}{10}
	\randint[Z]{ee}{-10}{10}
	\randint{h}{-10}{10}
	\randint[Z]{g}{-10}{10}
	
	
	\function[normalize]{f}{(a*x^2+b*x+c)/(ee*x^2+h*x+g)}
	
	\function[calculate]{sol}{a/ee}
	
\end{variables}

	\type{input.function}
	\field{rational} 
    \lang{de}{
	    \text{
	     Bestimmen Sie den Grenzwert $g$ für $x\to \infty$ für 
$ f:\R\to\R, x\mapsto \var{f}. $ Schreiben Sie $infty$ bzw. $-infty$, falls die Funktion
bestimmt gegen unendlich bzw. minus unendlich divergiert.
    }}
    
    \begin{answer}
    \text{Der Grenzwert $g$ ist gegeben durch $g$ = }
	    \solution{sol}
        \checkAsFunction{x}{-10}{10}{10}
        \explanation[edited]{Zählergrad = Nennergrad, also nicht bestimmt divergent.}
    \end{answer}
    
\end{question}
