%######################################################QUESTION_START	
	\begin{question}	
	%+++++++++++++++++++VARIABLES++++++++++++++++++++++
		\begin{variables}
			\randint{s}{1}{5}
			\randint{r}{1}{5}
			\randint{w1}{0}{1}
			\randint{w2}{0}{1}
			\function[calculate]{c}{s+1}
			
			\function[calculate]{a}{2*c+r+w2}
			\function[calculate]{b}{r-c^2+a*c+w1}
			
			\function{func1}{x^2-a*x+b}
			\function{func2}{r/(x-s)}
			\function{sol1}{w1}
			\function{sol2}{(1-w1)*w2}
			\function{sol3}{(1-w1)*(1-w2)}
			
		\end{variables}
	%+++++++++++++++++++TYPE+++++++++++++++++++++++++++	
	    \type{mc.unique}  %mc.yesno %mc.multiple
	%+++++++++++++++++++TITLE++++++++++++++++++++++++++	
	    \lang{de}{\text{Gegeben ist die Funktion $f:\R\to \R$ mit $f(x)=\begin{cases}\var{func1}, & \text{ für }x<\var{c}\\ 
     \var{func2}& \text{ für }x\geq \var{c}.\end{cases}$\\
     Welche Eigenschaften treffen auf die Funktion $f(x)$ an der Stelle $x_0=\var{c}$ zu?}}
     \lang{en}{\text{Given the function $f:\R\to \R$ with $f(x)=\begin{cases}\var{func1}, & \text{ for }x<\var{c}\\
     \var{func2}& \text{ for }x\geq \var{c}.\end{cases}$\\
     Which properties are true for the function $f(x)$ at the point $x_0=\var{c}$?}}
	%+++++++++++++++++++ANSWERS++++++++++++++++++++++++    
	    \begin{choice}
	    	\lang{de}{\text{weder stetig noch differenzierbar}}
        \lang{en}{\text{neither continuous nor differentiable}}
			\solution{compute}
			\iscorrect{sol1}{=}{1}
	    \end{choice}     
	    \begin{choice}
	    	\lang{de}{\text{stetig, aber nicht differenzierbar}}
        \lang{en}{\text{continuous but not differentiable}}
			\solution{compute}
			\iscorrect{sol2}{=}{1}
	    \end{choice}     
	    \begin{choice}
	    	\lang{de}{\text{differenzierbar, aber nicht stetig}}
        \lang{en}{\text{differentiable but not continuous}}
			\solution{false}
	    \end{choice}      
	    \begin{choice}
	    	\lang{de}{\text{sowohl stetig, als auch differenzierbar}}
        \lang{en}{\text{both continuous and differentiable}}
			\solution{compute}
			\iscorrect{sol3}{=}{1}
	    \end{choice}    
	\end{question}
%######################################################QUESTION_END


\end{problem}
