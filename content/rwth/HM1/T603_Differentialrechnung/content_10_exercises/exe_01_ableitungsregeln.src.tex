\lang{en}{
\begin{table}[\class{items}]
\nowrap{a) Find the derivative of the function $f(x)=xe^{x}$.} \\
\nowrap{b) Find the derivative of $f(x)=x^{4}+\ln(x)\cdot \sqrt{x}$.} \\
\nowrap{c) Find the derivative of $f(x)=\sin(x)\cos(x)$.}
\end{table}
}

\begin{tabs*}[\initialtab{0}\class{exercise}]
 \tab{
  \lang{de}{Antwort}
  \lang{en}{Answer}
  }
  \begin{table}[\class{items}]
\nowrap{a) $f'(x)=e^{x}(1+x)$} \\
\nowrap{b) $f'(x)=4x^{3}+\frac{\sqrt{x}}{x}+\frac{\ln(x)}{2\sqrt{x}}$} \\
\nowrap{c) $f'(x)=\cos^{2}(x)-\sin^{2}(x)$}
\end{table}
  
  \tab{
  \lang{de}{Lösung a)}
  \lang{en}{Solution a)}
  }
  
  \begin{incremental}[\initialsteps{1}]
    \step \lang{de}{Wir verwenden die Ableitungsformeln. Die Funktion $f$ lässt sich schreiben als Produkt $f(x)=u(x)v(x)$ mit $u(x)=x$ und $v(x)=e^{x}$.}
    \lang{en}{We will use the derivative rules. The function $f$ can be written as a product $f(x)=u(x)v(x)$ with $u(x)=x$ and $v(x)=e^{x}$.}
    \step \lang{de}{Die Produktregel besagt nun, dass $f'(x)=u'(x)v(x)+u(x)v'(x)$ für alle $x\in\R$ gilt. 
    Somit bekommen wir als Ableitung
\[f'(x)=e^{x}+xe^{x}=e^x(1+x)\,.\]}
\lang{en}{The product rule tells us that $f'(x)=u'(x)v(x)+u(x)v'(x)$ for all $x\in\R$. 
    Therefore, the derivative is
\[f'(x)=e^{x}+xe^{x}=e^x(1+x)\,.\]}
  \end{incremental}
  
  \tab{
  \lang{de}{Lösung b)}
  \lang{en}{Solution b)}
  }
  
  \begin{incremental}[\initialsteps{1}]
    \step \lang{de}{Wir verwenden wieder die Produktregel mit $u(x)=\ln(x)$ und $v(x)=\sqrt{x}$ für alle $x>0$.}
    \lang{en}{We will use the product rule again, but now with $u(x)=\ln(x)$ and $v(x)=\sqrt{x}$ for $x>0$.}
    \step \lang{de}{Aus der Linearität der Ableitung ergibt sich somit 
\[f'(x)=\frac{d}{dx}(x^{4})+u'(x)v(x)+u(x)v'(x)=4x^{3}+\frac{\sqrt{x}}{x}+\frac{\ln(x)}{2\sqrt{x}}\]
für alle $x>0$ als Ableitung.}
\lang{en}{Since the derivative is linear, we find
\[f'(x)=\frac{d}{dx}(x^{4})+u'(x)v(x)+u(x)v'(x)=4x^{3}+\frac{\sqrt{x}}{x}+\frac{\ln(x)}{2\sqrt{x}}\]
for $x>0$.}
  \end{incremental}  

 \tab{
  \lang{de}{Lösung c)}
  \lang{en}{Solution c)}
  }
  
 \lang{de}{Wir setzen $u(x)=\sin(x)$ und $v(x)=\cos(x)$. Aus der Produktregel erhalten wir mit Hilfe der Ableitungsformeln für Sinus und Cosinus
\[f'(x)=u'(x)v(x)+u(x)v'(x)=\cos(x)\cos(x)+\sin(x)[-\sin(x)]=\cos^{2}(x)-\sin^{2}(x)\]
für alle $x\in\R$.} 
\lang{en}{Let $u(x)=\sin(x)$ and $v(x)=\cos(x)$. The product rule, together with the formulas for the derivative of sine and cosine, yield
\[f'(x)=u'(x)v(x)+u(x)v'(x)=\cos(x)\cos(x)+\sin(x)[-\sin(x)]=\cos^{2}(x)-\sin^{2}(x)\]
for all $x\in\R$.}
