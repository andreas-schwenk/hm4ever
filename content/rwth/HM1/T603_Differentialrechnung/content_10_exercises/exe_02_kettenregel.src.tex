\lang{de}{Berechnen Sie die Ableitungen der folgenden Funktionen mit Hilfe der Kettenregel. Geben Sie auch ihre Definitionsbereiche an.}

\lang{de}{Find the derivative of the following functions using the chain rule. Include the domain of definition.}


\begin{table}[\class{items}]
\nowrap{a) $f(x)=\exp(\sin(2x))$} \\
\nowrap{b) $f(x)=\ln(x^{2}+1)$} \\
\nowrap{c) $f(x)=\cos^{2}(\ln(x+1))$}
\end{table}

\begin{tabs*}[\initialtab{0}\class{exercise}]
  \tab{
  \lang{de}{Lösung a)}
  \lang{en}{Solution a)}
  }
  
  \lang{de}{Die Funktion $f$ ist auf ganz $\R$ definiert. Mit Hilfe der Kettenregel erhalten wir als Ableitung}
  \lang{en}{The function $f$ is defined on all of $\R$. Using the chain rule, we find the derivative}
\[f'(x)=\exp'(\sin(2x))\cdot \sin'(2x)\cdot (2x)'=\exp(\sin(2x))\cos(2x)\cdot 2=2\exp(\sin(2x))\cos(2x)\,.\]

\lang{de}{Die Funktion $f$ ist auf ganz $\R$ definiert. Mit Hilfe der Kettenregel erhalten wir als Ableitung}
\lang{en}{The function $f$ is defined on all of $\R$. Using the chain rule, we find the derivative}
\[f'(x)=\ln'(x^{2}+1)\cdot (x^{2}+1)'=\frac{2x}{x^{2}+1}\,.\]

	\lang{de}{Die Funktion $f$ ist überall dort definert, wo $x+1$ postiv ist. Daher lautet der Definitionsbereich von $f$ genau $(-1,\infty)$.  Mit Hilfe der Kettenregel erhalten wir als Ableitung}
 \lang{en}{The function $f$ is defined wherever $x+1$ is positive. Therefore, the domain of definition of $f$ is exactly $(-1, \infty)$. Using the chain rule, we find the derivative }
\begin{align*}
   f'(x)&=&&2\cos(\ln(x+1))\cdot \cos'(\ln(x+1))\cdot(\ln'(x+1))\cdot (x+1)'&\\
&=&&2\cos(\ln(x+1))(-\sin(\ln(x+1)))\cdot\frac{1}{x+1}\,&\\
&=&&\frac{-2}{x+1}\cos(\ln(x+1))\sin(\ln(x+1))\,.&
  \end{align*}
