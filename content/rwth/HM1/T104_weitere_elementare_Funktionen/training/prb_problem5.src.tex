\begin{question}
      \lang{de}{\text{W\"{a}hlen Sie alle richtigen Antworten aus.}}
      \lang{en}{\text{Select all of the correct answers.}}
      \explanation{}

      \permutechoices{1}{3}
      \type{mc.multiple}
      \field{real}

      \begin{variables}
            \randint{a}{1}{11}
            \randint{b}{2}{12}
             
            \randint{c}{2}{4}
             \function[calculate]{d}{-c}
              \function[calculate]{m}{c+b}
              \function[calculate]{n}{-c+b}
                \randadjustIf{a,b}{a=b} % see Part 9 http://team.mumie.net/projects/support/wiki/GenericTexProblems 
      \end{variables}
      \begin{choice}
            \text{$(\var{a} \cdot \var{b})^{\var{c}}=\var{a}^{\var{c}}\cdot \var{b}^{\var{c}}$}
            \solution{true}
      \end{choice}

\begin{choice}
            \text{$\var{a}^{\var{c}} \cdot \var{a}^{\var{b}} =\var{a}^{\var{m}}$}
            \solution{true}
      \end{choice}
      \begin{choice}
            \text{$\var{a}^{-\var{c}} \cdot \var{a}^{\var{b}} =\var{a}^{\var{n}}$}
            \solution{true}
      \end{choice}
           
     \end{question}


\begin{question}
      \lang{de}{\text{W\"{a}hlen Sie alle richtigen Antworten aus.}}
      \lang{en}{\text{Select all of the correct answers.}}
      \explanation{}

      \permutechoices{1}{3}
      \type{mc.multiple}
      \field{real}

      \begin{variables}
            \randint{a}{3}{11}
            \randint{b}{2}{12}
            
            \randint{c}{2}{4}
             \function[calculate]{d}{-c}
              \function[calculate]{m}{c-b}
              \function[calculate]{n}{-c-b}
                \randadjustIf{a,b}{a=b} % see Part 9 http://team.mumie.net/projects/support/wiki/GenericTexProblems 
      \end{variables}
      \begin{choice}
            \lang{de}{\text{$(\var{a} : \var{b})^{\var{c}}=\var{a}^{\var{c}}: \var{b}^{\var{c}}$}}
            \lang{en}{\text{$(\var{a} / \var{b})^{\var{c}}=\var{a}^{\var{c}}/ \var{b}^{\var{c}}$}}
            \solution{true}
      \end{choice}

\begin{choice}
            \lang{de}{\text{$\var{a}^{\var{c}} : \var{a}^{\var{b}} =\var{a}^{\var{m}}$}}
            \lang{en}{\text{$\var{a}^{\var{c}} / \var{a}^{\var{b}} =\var{a}^{\var{m}}$}}
            \solution{true}
      \end{choice}
      \begin{choice}
            \lang{de}{\text{$\var{a}^{-\var{c}} : \var{a}^{-\var{b}} =\var{a}^{\var{n}}$}}
            \lang{en}{\text{$\var{a}^{-\var{c}} / \var{a}^{-\var{b}} =\var{a}^{\var{n}}$}}
            \solution{false}
      \end{choice}
     \end{question}

    \begin{question}
      \lang{de}{\text{W\"{a}hlen Sie alle richtigen Antworten aus.}}
      \lang{en}{\text{Select all of the correct answers.}}
      \explanation{}

      \permutechoices{1}{4}
      \type{mc.multiple}
      \field{real}

      \begin{variables}
            \randint{a}{3}{25}
            \randint{b}{2}{35}
            \randint{m}{2}{9}
             \function[calculate]{c}{a+b}
              \function[calculate]{d}{b*a}
                          \randadjustIf{a,b}{a=b} % see Part 9 http://team.mumie.net/projects/support/wiki/GenericTexProblems 
      \end{variables}
      \begin{choice}
            \text{$\var{m}^{\var{a}} \cdot \var{m}^{\var{b}}=\var{m}^{\var{c}}$}
            \solution{true}
      \end{choice}

      \begin{choice}
            \text{$\var{m}^{\var{a}} + \var{m}^{\var{b}}=\var{m}^{\var{c}}$}
            \solution{false}
      \end{choice}
            \begin{choice}
            \text{$\var{m}^{\var{a}} \cdot \var{m}^{\var{b}}=\var{m}^{\var{d}}$}
            \solution{false}
      \end{choice}
      \begin{choice}
            \text{$\left(\var{m}^{\var{a}}\right)^{\var{b}}=\var{m}^{\var{d}}$}
            \solution{true}
      \end{choice}
\end{question}
