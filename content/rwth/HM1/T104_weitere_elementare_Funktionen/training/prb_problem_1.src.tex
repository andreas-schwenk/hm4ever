    \begin{question}
      \lang{de}{\text{W\"{a}hlen Sie alle richtigen Antworten aus.}}
      \lang{en}{\text{Choose all the correct answers.}}
      \explanation{}

      \permutechoices{1}{5}
      \type{mc.yesno}
      \field{real} 
 
      \begin{variables}
            \randint{a}{20}{30}
            \randint{b}{6}{11}
             \function[calculate]{c}{a+2}
              \function[calculate]{d}{b-2}
              \function[calculate]{ee}{-a+1}
              \function[calculate]{f}{-b-5}
      \end{variables}
      \begin{choice}
            \text{$\var{a} \le \var{a}$}
            \solution{true}
      \end{choice}

      \begin{choice}
            \text{$-\var{c} \ge \var{d}$}
            \solution{false}
      \end{choice}
             \begin{choice}
            \text{$\var{ee} \le -\var{d}$}
            \solution{true}
      \end{choice}
             \begin{choice}
            \text{$\var{ee} > \var{f}$}
            \solution{false}
      \end{choice}
      \begin{choice}
           \text{$ 0 = -0$}
            \solution{true} % the correct answer will be calculate automatically. 
             %The use of an additional \iscorrect command is required
      \end{choice}
\end{question}

\begin{question}
       \lang{de}{\text{W\"{a}hlen Sie alle richtigen Antworten aus.}}
      \lang{en}{\text{Choose all the correct answers.}}
      \explanation{}
      \permutechoices{1}{3}
      \type{mc.yesno}
      \field{real}
    \begin{variables}
            \randint{b}{11}{19}
            \randint{a}{2}{9}

            \function{f}{a+b}                               
            \function[calculate]{f_result}{a+b}

            \function{g}{-a+b}
            \function[calculate]{g_result}{-a+b}

            \function{h}{(-a)+(-b)}
            \function[calculate]{h_result}{(-a)+(-b)+1}
            
            

      \end{variables}

      \begin{choice}
            \text{$\var{f} = \var{f_result}$}
            \solution{true} % This choice is correct
      \end{choice}

      \begin{choice}
            \text{$\var{g} = \var{g_result}$}
            \solution{true} % This choice is wrong
      \end{choice}

      \begin{choice}
            \text{$\var{h} = \var{h_result}$}
            \solution{false} % This choice is wrong
      \end{choice}
\end{question}
