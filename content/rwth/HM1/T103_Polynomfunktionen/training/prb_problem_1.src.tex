\begin{problem}
 	\begin{variables}
      		\randint{a0}{-5}{5}
            \randint{a1}{-5}{5}
            \randint{a2}{-5}{5}
            \randint{a3}{-5}{5}
            \randint[Z]{a4}{-5}{5}
            \function[expand, normalize]{f1}{a4*x^4+ a3*x^3+a2*x^2+a1*x+a0}
            
            \randint{b0}{-5}{5}
            \randint{b1}{-5}{5}
            \randint{b2}{-5}{5}
            \randint[Z]{b3}{-5}{5}
            \function[expand, normalize]{f2}{b3*x^3+b2*x^2+b1*x+b0}
            
            \number{s1}{4}
            \number{s2}{3}
    \end{variables}
     
    \begin{question}
    	\type{input.number}
		\field{rational}
		\lang{de}{\text{Tragen Sie den Leitkoeffizienten und den Grad des folgenden Polynoms ein:\\
		$p(x)= \var{f1}$}}
      \lang{en}{\text{}}
      \explanation{}
     
      
		\begin{answer}
		    \text{Leitkoeffizient:}
		    \solution{a4}
		\end{answer}
      	\begin{answer}
		    \text{Grad des Polynoms:}
		    \solution{s1}
            \explanation{Der Leitkoeffizient ist der Koeffizient des Monoms höchsten Grades. }
		\end{answer}
        
	\end{question}

	\begin{question}
    	\type{input.number}
		\field{rational} 
      \lang{de}{\text{Tragen Sie den Leitkoeffizienten und den Grad des folgenden Polynoms ein:\\
		$p(x)= \var{f2}$}}
      \lang{en}{\text{}}
      \explanation{Der Leitkoeffizient ist der Koeffizient des Monoms höchsten Grades.}

		\begin{answer}
		    \text{Leitkoeffizient:}
		    \solution{b3}
		\end{answer}
      	\begin{answer}
		    \text{Grad des Polynoms:}
		    \solution{s2}
		\end{answer}
	\end{question}
   
\end{problem}
