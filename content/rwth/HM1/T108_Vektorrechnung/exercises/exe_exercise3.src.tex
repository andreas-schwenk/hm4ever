 \begin{tabs*}[\initialtab{0}\class{exercise}]
  \tab{\lang{de}{Lösung 1.}\lang{en}{Solution 1.}}

    \begin{displaymath}
    \begin{mtable}[\cellaligns{lccccccc}]
    \text{a)}& \begin{pmatrix} 2\\-1 \\3\end{pmatrix} &+& \begin{pmatrix} -3\\-2\\-1\end{pmatrix} &=& \begin{pmatrix} 2+(-3)\\-1+(-2) \\3+(-1)\end{pmatrix} &=& \begin{pmatrix} -1\\-3 \\2\end{pmatrix}\\
&&&&&&&\\

    \text{b)}& \begin{pmatrix} 5\\2\\-4\end{pmatrix} &+& \begin{pmatrix} 1\\2\\1\end{pmatrix} &=& \begin{pmatrix} 5+1\\2+2\\-4+1\end{pmatrix} &=& \begin{pmatrix} 6\\4\\-3\end{pmatrix}\\
&&&&&&&\\
    
    \text{c)}& \begin{pmatrix} 1\\1\\-1\end{pmatrix} &+& \begin{pmatrix} 2\\1\\-1\end{pmatrix} &=& \begin{pmatrix} 1+2\\1+1\\-1+(-1)\end{pmatrix} &=& \begin{pmatrix} 3\\2\\-2\end{pmatrix}\\
&&&&&&&\\
    
    \text{d)}& \begin{pmatrix} 3\\2\\4\end{pmatrix} &+& \begin{pmatrix} -2\\2\\-2\end{pmatrix} &=& \begin{pmatrix} 3+(-2)\\2+2\\4+(-2)\end{pmatrix} &=& \begin{pmatrix} 1\\4\\2\end{pmatrix}\\
    \end{mtable}
    \end{displaymath}
        


  \tab{\lang{de}{Lösung 2.}\lang{en}{Solution (2)}}

    \begin{displaymath}
    \begin{mtable}[\cellaligns{lrcrcr}]
    \text{a)}& 3\cdot \begin{pmatrix} 2\\-2 \\-1\end{pmatrix} &=& \begin{pmatrix} 3\cdot 2\\3\cdot(-2) \\3\cdot(-1)\end{pmatrix} &=& \begin{pmatrix} 6\\-6 \\-3\end{pmatrix}\\
&&&&&\\

    \text{b)}&-2\cdot \begin{pmatrix} 1\\2\\-4\end{pmatrix} &=& \begin{pmatrix} -2\cdot 1\\-2\cdot 2\\-2\cdot(-4)\end{pmatrix} &=& \begin{pmatrix} -2\\-4\\8\end{pmatrix}\\
&&&&&\\
    
    \text{c)}&-5\cdot \begin{pmatrix} 1\\1\\-1\end{pmatrix} &=& \begin{pmatrix} -5\cdot 1\\-5\cdot 1\\-5\cdot(-1)\end{pmatrix} &=& \begin{pmatrix} -5\\-5\\5\end{pmatrix}\\
&&&&&\\
    
    \text{d)}& 2\cdot\begin{pmatrix} -2\\2\\-2\end{pmatrix} &=& \begin{pmatrix} 2\cdot(-2)\\2\cdot 2\\2\cdot(-2)\end{pmatrix} &=& \begin{pmatrix} -4\\4\\-4\end{pmatrix}\\
    \end{mtable}
    \end{displaymath}
        
\end{tabs*}
