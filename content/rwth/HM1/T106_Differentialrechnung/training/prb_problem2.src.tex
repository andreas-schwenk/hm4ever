    \begin{question}
      \lang{de}{\text{Bestimmen Sie die Ableitung von $f(x)=\var{f1}$.}}
      \type{input.function}
 
      \begin{variables}
            \randint{n}{1}{6}
            \function[expand,normalize]{f1}{sin(x)*x^n+e^x}
            \function{s1}{cos(x)*x^n+n*sin(x)*x^(n-1)+e^x}
      \end{variables}
      
      \begin{answer}
      		\text{$f'(x)=$}
			\solution{s1}
			\checkAsFunction{x}{-10}{10}{100}
            \explanation{Hier bietet sich die Produktregel an.}
		\end{answer}
\end{question}
